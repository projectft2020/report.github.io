## 1. vectorBT 動態計算能力分析

### 1.1 vectorBT 內建的統計功能

vectorBT 提供了完整的統計計算能力，包括：

**Portfolio.stats() 方法**:
```python
import vectorbt as vbt

# 假設已有回測的 portfolio 對象
pf = vbt.Portfolio.from_signals(...)

# 獲取完整統計信息
stats = pf.stats()

# stats 包含以下關鍵指標：
# - Win Rate [%]        : 勝率（盈利交易 / 總交易）
# - Profit Factor       : 賺賠比（總盈利 / 總虧損）
# - Expectancy          : 期望值
# - Total Trades        : 總交易數
# - Total Closed Trades : 已平倉交易數
```

**Positions.stats() 方法**:
```python
# 也可以從 positions 對象獲取統計
positions_stats = pf.positions.stats()
```

**Trades.stats() 方法**:
```python
# 從交易記錄獲取統計
trades_stats = pf.trades.stats()
```

### 1.2 計算公式

vectorBT 使用以下公式計算這些指標：

**勝率 (Win Rate)**:
```
Win Rate = (盈利交易數 / 總已平倉交易數) × 100%
```

**賺賠比 (Profit Factor)**:
```
Profit Factor = 總盈利金額 / 總虧損金額
```

### 1.3 數據來源

這些統計基於 vectorBT 的 **Trade Records**：
- 每筆交易都有 PnL（盈虧）信息
- 包含進場價格、出場價格、費用等信息
- 可以區分已平倉和未平倉交易

function calculateMetrics(trades: Trade[]) {
  const winningTrades = trades.filter(t => t.pnl > 0);
  const losingTrades = trades.filter(t => t.pnl < 0);

  const winRate = (winningTrades.length / trades.length) * 100;
  const totalProfit = winningTrades.reduce((sum, t) => sum + t.pnl, 0);
  const totalLoss = Math.abs(losingTrades.reduce((sum, t) => sum + t.pnl, 0));
  const profitFactor = totalLoss > 0 ? totalProfit / totalLoss : Infinity;

  return { winRate, profitFactor };
}

# utils/backtest_metrics.py
import pandas as pd
import numpy as np
import json
from typing import Tuple, Optional, Dict, Any

def calculate_win_rate_profit_factor(
    history_record: Dict[str, Any]
) -> Tuple[Optional[float], Optional[float]]:
    """
    從 backtest_history 記錄動態計算 win_rate 和 profit_factor

    Args:
        history_record: backtest_history 表的一條記錄（字典格式）

    Returns:
        tuple: (win_rate, profit_factor) 如果無法計算返回 (None, None)
    """

    # 嘗試從 trades_json 字段計算
    if 'trades_json' in history_record and history_record['trades_json']:
        try:
            trades = json.loads(history_record['trades_json'])
            trades_df = pd.DataFrame(trades)

            return _calculate_from_dataframe(trades_df)
        except (json.JSONDecodeError, KeyError, ValueError) as e:
            print(f"Error parsing trades_json: {e}")
            return None, None

    # 嘗試從 params 中的 vectorBT 結果計算
    if 'params' in history_record and history_record['params']:
        try:
            params = json.loads(history_record['params'])
            if 'vectorbt_portfolio' in params:
                return _calculate_from_vectorbt(params['vectorbt_portfolio'])
        except (json.JSONDecodeError, KeyError) as e:
            print(f"Error parsing params: {e}")

    # 嘗試從單獨的字段組合計算
    if all(key in history_record for key in ['winning_trades', 'losing_trades', 'total_profit', 'total_loss']):
        winning_count = history_record['winning_trades']
        losing_count = history_record['losing_trades']
        total_profit = history_record['total_profit']
        total_loss = history_record['total_loss']

        if (winning_count + losing_count) > 0:
            win_rate = (winning_count / (winning_count + losing_count)) * 100
        else:
            win_rate = None

        if total_loss > 0:
            profit_factor = total_profit / total_loss
        else:
            profit_factor = float('inf') if total_profit > 0 else None

        return win_rate, profit_factor

    return None, None

def _calculate_from_dataframe(trades_df: pd.DataFrame) -> Tuple[float, float]:
    """
    從交易 DataFrame 計算 win_rate 和 profit_factor
    """
    if trades_df.empty:
        return None, None

    # 檢查必需字段
    required_fields = ['pnl']
    if not all(field in trades_df.columns for field in required_fields):
        print(f"Missing required fields: {[f for f in required_fields if f not in trades_df.columns]}")
        return None, None

    # 過濾掉 PnL 為 0 的交易（如果需要）
    filtered_df = trades_df[trades_df['pnl'] != 0]

    if filtered_df.empty:
        return None, None

    # 勝率 = 盈利交易數 / 總交易數
    winning_trades = filtered_df[filtered_df['pnl'] > 0]
    total_trades = len(filtered_df)

    win_rate = (len(winning_trades) / total_trades) * 100

    # 賺賠比 = 總盈利 / 總虧損
    total_profit = winning_trades['pnl'].sum()
    losing_trades = filtered_df[filtered_df['pnl'] < 0]
    total_loss = abs(losing_trades['pnl'].sum())

    if total_loss > 0:
        profit_factor = total_profit / total_loss
    else:
        profit_factor = float('inf') if total_profit > 0 else 0.0

    return win_rate, profit_factor

def _calculate_from_vectorbt(serialized_pf: str) -> Tuple[float, float]:
    """
    從序列化的 vectorBT Portfolio 對象計算
    """
    try:
        # 這裡需要根據實際的序列化方式進行反序列化
        # 例如使用 pickle, json, 或其他格式
        import pickle
        pf = pickle.loads(serialized_pf.encode('latin1'))

        # 使用 vectorBT 的 stats() 方法
        stats = pf.stats()

        win_rate = stats.get('Win Rate [%]', None)
        profit_factor = stats.get('Profit Factor', None)

        return win_rate, profit_factor
    except Exception as e:
        print(f"Error deserializing vectorBT portfolio: {e}")
        return None, None

# 添加緩存裝飾器
from functools import lru_cache

@lru_cache(maxsize=100)
def calculate_win_rate_profit_factor_cached(
    history_id: int,
    trades_json: str
) -> Tuple[Optional[float], Optional[float]]:
    """
    帶緩存的計算函數
    """
    history_record = {
        'trades_json': trades_json,
        'id': history_id
    }
    return calculate_win_rate_profit_factor(history_record)
