```python
class RSIStrategy(IStrategy):
    """Mean reversion strategy using RSI indicator"""

    name = "rsi_mean_reversion"

    def __init__(
        self,
        period: int = 14,
        buy_threshold: int = 30,
        sell_threshold: int = 70,
        min_trade_size: float = 1000.0
    ):
        if not (0 < buy_threshold < sell_threshold < 100):
            raise ValueError("Invalid threshold values")
        self.period = period
        self.buy_threshold = buy_threshold
        self.sell_threshold = sell_threshold
        self.min_trade_size = min_trade_size
        self.rsi_cache = {}

    def generate_signals(
        self,
        context: ExecutionContext,
        symbols: List[str]
    ) -> List[StrategySignal]:
        signals = []

        for symbol in symbols:
            # Check market status
            if context.market_status != 'CLOSED':
                continue

            # Check if already holding
            existing_position = next(
                (p for p in context.current_positions if p.symbol == symbol),
                None
            )

            # Get RSI value

            # Generate buy signal (only if not holding)
            if rsi < self.buy_threshold and not existing_position:
                signals.append(StrategySignal(
                    symbol=symbol,
                    action=SignalAction.BUY,
                    confidence=self._calculate_confidence(rsi, self.buy_threshold)

            # Generate sell signal (only if holding)
            elif rsi > self.sell_threshold and existing_position:
                signals.append(StrategySignal(
                    symbol=symbol,
                    action=SignalAction.SELL,
                    confidence=self._calculate_confidence(rsi, self.sell_threshold)
```

```python
class ConfluenceStrategy(IStrategy):
    """Strategy requiring multiple indicators to agree"""

    name = "multi_indicator_confluence"

    def __init__(self, min_agreement: int = 2):
        self.min_agreement = min_agreement  # Number of indicators required to agree
        self.rsi_strategy = RSIStrategy()
        self.macd_strategy = MACDStrategy()
        self.indicator_cache = {}

    def generate_signals(
        self,
        context: ExecutionContext,
        symbols: List[str]
    ) -> List[StrategySignal]:
        signals = []

        for symbol in symbols:
            # Get signals from each sub-strategy
            rsi_signals = self.rsi_strategy.generate_signals(context, [symbol])
            macd_signals = self.macd_strategy.generate_signals(context, [symbol])

            # Count buy signals
            buy_votes = sum(
                1 for s in rsi_signals + macd_signals
                if s.action == SignalAction.BUY
            )

            # Count sell signals
            sell_votes = sum(
                1 for s in rsi_signals + macd_signals
                if s.action == SignalAction.SELL
            )

            # Generate buy signal if minimum agreement reached
            if buy_votes >= self.min_agreement:
                avg_confidence = (
                    sum(s.confidence for s in rsi_signals + macd_signals if s.action == SignalAction.BUY)
                    / buy_votes
                )
                signals.append(StrategySignal(
                    symbol=symbol,
                    action=SignalAction.BUY,
                    confidence=avg_confidence
                ))

            # Generate sell signal if minimum agreement reached
            elif sell_votes >= self.min_agreement:
                avg_confidence = (
                    sum(s.confidence for s in rsi_signals + macd_signals if s.action == SignalAction.SELL)
                    / sell_votes
                )
                signals.append(StrategySignal(
                    symbol=symbol,
                    action=SignalAction.SELL,
                    confidence=avg_confidence
                ))

        return signals
```

### 1.5 Strategy Template: Composite Strategy

```python
class CompositeStrategy(IStrategy):
    """Combines multiple strategies with configurable merge algorithm"""

    name = "composite_strategy"

    MERGE_ALGORITHMS = [
        'majority_vote',  # Buy if majority says buy
        'unanimous',      # Buy only if all say buy
        'weighted',       # Weight by confidence
        'any'             # Buy if any says buy
    ]

    def __init__(
        self,
        strategies: List[IStrategy],
        merge_algorithm: str = 'majority_vote'
    ):
        if merge_algorithm not in self.MERGE_ALGORITHMS:
            raise ValueError(f"Invalid merge algorithm: {merge_algorithm}")
        self.strategies = strategies
        self.merge_algorithm = merge_algorithm

    def generate_signals(
        self,
        context: ExecutionContext,
        symbols: List[str]
    ) -> List[StrategySignal]:
        # Collect all signals from all strategies
        all_signals_by_symbol = {symbol: [] for symbol in symbols}

        for strategy in self.strategies:
            strategy_signals = strategy.generate_signals(context, symbols)
            for signal in strategy_signals:
                all_signals_by_symbol[signal.symbol].append(signal)

        # Merge signals for each symbol
        merged_signals = []
        for symbol, signals in all_signals_by_symbol.items():
            if not signals:
                continue

            merged_signal = self._merge_signals(symbol, signals)
            if merged_signal:
                merged_signals.append(merged_signal)

        return merged_signals

    def _merge_signals(self, symbol: str, signals: List[StrategySignal]) -> StrategySignal:
        """Merge multiple signals using configured algorithm"""

        if self.merge_algorithm == 'majority_vote':
            buy_count = sum(1 for s in signals if s.action == SignalAction.BUY)
            sell_count = sum(1 for s in signals if s.action == SignalAction.SELL)

            if buy_count > sell_count:
                return StrategySignal(
                    symbol=symbol,
                    action=SignalAction.BUY,
                    confidence=buy_count / len(signals)
                )
            elif sell_count > buy_count:
                return StrategySignal(
                    symbol=symbol,
                    action=SignalAction.SELL,
                    confidence=sell_count / len(signals)
                )

        elif self.merge_algorithm == 'unanimous':
            actions = [s.action for s in signals]
            if all(a == SignalAction.BUY for a in actions):
                return StrategySignal(
                    symbol=symbol,
                    action=SignalAction.BUY,
                    confidence=1.0
                )
            elif all(a == SignalAction.SELL for a in actions):
                return StrategySignal(
                    symbol=symbol,
                    action=SignalAction.SELL,
                    confidence=1.0
                )

        elif self.merge_algorithm == 'weighted':
            buy_weight = sum(s.confidence for s in signals if s.action == SignalAction.BUY)
            sell_weight = sum(s.confidence for s in signals if s.action == SignalAction.SELL)

            if buy_weight > sell_weight:
                return StrategySignal(
                    symbol=symbol,
                    action=SignalAction.BUY,
                    confidence=min(1.0, buy_weight / (buy_weight + sell_weight))
                )
            elif sell_weight > buy_weight:
                return StrategySignal(
                    symbol=symbol,
                    action=SignalAction.SELL,
                    confidence=min(1.0, sell_weight / (buy_weight + sell_weight))
                )

        elif self.merge_algorithm == 'any':
            buy_signals = [s for s in signals if s.action == SignalAction.BUY]
            sell_signals = [s for s in signals if s.action == SignalAction.SELL]

            if buy_signals:
                max_conf = max(s.confidence for s in buy_signals)
                return StrategySignal(
                    symbol=symbol,
                    action=SignalAction.BUY,
                    confidence=max_conf
                )
            elif sell_signals:
                max_conf = max(s.confidence for s in sell_signals)
                return StrategySignal(
                    symbol=symbol,
                    action=SignalAction.SELL,
                    confidence=max_conf
                )

        return None
```

```python
import pandas as pd
from typing import List
from datetime import date

class VBTDataLoader:
    """Load and format data for VectorBT execution"""

    def __init__(self, connection):
        self.conn = connection
        self.data_cache = {}

    def load_ohlcv(
        self,
        symbols: List[str],
        start_date: date,
        end_date: date
    ) -> pd.DataFrame:
        """
        Load OHLCV data in VectorBT format.

        Returns: MultiIndex DataFrame with (symbol, date) as index
        """
        cache_key = (tuple(symbols), start_date, end_date)
        if cache_key in self.data_cache:
            return self.data_cache[cache_key]

        query = """
        SELECT
            symbol,
            date,
            open,
            high,
            low,
            close,
            volume
        FROM prices
        WHERE symbol IN ({placeholders})
          AND date BETWEEN ? AND ?
        ORDER BY symbol, date
        """.format(placeholders=','.join(['?'] * len(symbols)))

        params = symbols + [start_date, end_date]
        df = pd.read_sql(query, self.conn, params=params)

        # Format for VectorBT: MultiIndex (symbol, date)
        df = df.set_index(['symbol', 'date'])

        self.data_cache[cache_key] = df
        return df

    def load_prices(
        self,
        symbols: List[str],
        start_date: date,
        end_date: date
    ) -> pd.DataFrame:
        """
        Load close prices for portfolio creation.

        Returns: DataFrame with date index and symbol columns
        """
        ohlcv = self.load_ohlcv(symbols, start_date, end_date)
        return ohlcv['close'].unstack(level='symbol')
```

```python
import pandas as pd
from typing import Dict, List
from datetime import date

class SignalAdapter:
    """Convert IStrategy signals to VectorBT format"""

    @staticmethod
    def signals_to_vectorbt(
        ohlcv_index: pd.MultiIndex,
        signals_by_date: Dict[date, List[StrategySignal]]
    ) -> tuple[pd.DataFrame, pd.DataFrame]:
        """
        Convert strategy signals to VectorBT entry/exit DataFrames.

        Args:
            ohlcv_index: MultiIndex from VBTDataLoader
            signals_by_date: Dict mapping date to List[StrategySignal]

        Returns:
            entries: DataFrame[symbol] = bool (True = buy signal)
            exits: DataFrame[symbol] = bool (True = sell signal)
        """
        # Extract all symbols from index
        symbols = ohlcv_index.get_level_values('symbol').unique()

        # Create empty DataFrames
        entries = pd.DataFrame(False, index=ohlcv_index.get_level_values('date').unique(), columns=symbols)
        exits = pd.DataFrame(False, index=ohlcv_index.get_level_values('date').unique(), columns=symbols)

        # Populate signals
        for signal_date, signals in signals_by_date.items():
            if signal_date not in entries.index:
                continue

            for signal in signals:
                if signal.symbol not in symbols:
                    continue

                if signal.action == SignalAction.BUY:
                    entries.loc[signal_date, signal.symbol] = True
                elif signal.action == SignalAction.SELL:
                    exits.loc[signal_date, signal.symbol] = True

        return entries, exits
```

```python
from typing import Dict, List

class ExecutionContextFactory:
    """Create ExecutionContext instances with portfolio state management"""

    def __init__(self, initial_cash: float = 100000):
        self.initial_cash = initial_cash
        self.positions: Dict[str, PositionState] = {}

    def create_context(
        self,
        date: date,
        current_prices: Dict[str, float],
        cash: float
    ) -> ExecutionContext:
        """
        Create execution context for a given date.

        Args:
            date: Current market date
            current_prices: Dict mapping symbol to current price
            cash: Available cash

        Returns:
            ExecutionContext with current portfolio state
        """
        # Calculate current portfolio value
        position_value = sum(
            pos.quantity * current_prices.get(pos.symbol, pos.current_price)
            for pos in self.positions.values()
        )
        portfolio_value = cash + position_value

        # Update position current prices
        for symbol, price in current_prices.items():
            if symbol in self.positions:
                self.positions[symbol].current_price = price

        return ExecutionContext(
            portfolio_value=portfolio_value,
            available_cash=cash,
            current_positions=list(self.positions.values()),
            market_date=date,
            market_status='CLOSED'  # Simplified
        )

    def update_position(self, signal: StrategySignal, price: float):
        """Update position state based on signal"""
        if signal.action == SignalAction.BUY:
            self.positions[signal.symbol] = PositionState(
                symbol=signal.symbol,
                quantity=0,  # Will be calculated based on position sizing
                current_price=price
            )
        elif signal.action == SignalAction.SELL and signal.symbol in self.positions:
            del self.positions[signal.symbol]

    def reset(self):
        """Reset position state"""
        self.positions.clear()
```

```python
import vectorbt as vbt
import pandas as pd
from typing import List, Dict, Any
from datetime import date

class UnifiedStrategyEngine:
    """Core orchestrator integrating IStrategy with VectorBT"""

    def __init__(
        self,
        data_loader: VBTDataLoader,
        context_factory: ExecutionContextFactory
    ):
        self.data_loader = data_loader
        self.context_factory = context_factory

    def backtest(
        self,
        strategy: IStrategy,
        symbols: List[str],
        start_date: date,
        end_date: date,
        initial_cash: float = 100000
    ) -> Dict[str, Any]:
        """
        Execute backtest using VectorBT.

        Returns:
            Dict with performance metrics and results
        """
        # 1. Load historical data
        ohlcv = self.data_loader.load_ohlcv(symbols, start_date, end_date)
        prices = self.data_loader.load_prices(symbols, start_date, end_date)

        # 2. Generate daily signals
        dates = pd.date_range(start_date, end_date, freq='D')
        signals_by_date = self._generate_signals_daily(
            strategy,
            dates,
            symbols,
            start_date,
            end_date,
            initial_cash
        )

        # 3. Convert signals to VectorBT format
        entries, exits = SignalAdapter.signals_to_vectorbt(
            ohlcv.index,
            signals_by_date
        )

        # 4. Execute VectorBT backtest
        pf = vbt.Portfolio.from_signals(
            prices,
            entries=entries,
            exits=exits,
            init_cash=initial_cash,
            freq='1D',
            fees=0.001,
            slippage=0.0001
        )

        # 5. Extract metrics
        stats = pf.stats()

        return {
            'strategy': strategy.name,
            'symbols': symbols,
            'start_date': str(start_date),
            'end_date': str(end_date),
            'initial_cash': initial_cash,
            'final_value': float(pf.value().iloc[-1]),
            'total_return': float(stats['total_return']),
            'sharpe_ratio': float(stats['sharpe_ratio']),
            'max_drawdown': float(stats['max_drawdown']),
            'win_rate': float(stats['win_rate']),
            'num_trades': int(stats['num_trades']),
            'equity_curve': pf.value().tolist(),
            'entries': entries.to_dict(),
            'exits': exits.to_dict()
        }

    def _generate_signals_daily(
        self,
        strategy: IStrategy,
        dates: pd.DatetimeIndex,
        symbols: List[str],
        start_date: date,
        end_date: date,
        initial_cash: float
    ) -> Dict[date, List[StrategySignal]]:
        """
        Generate signals for each date in range.

        Returns:
            Dict mapping date to List[StrategySignal]
        """
        signals_by_date = {}
        cash = initial_cash

        # Reset context factory
        self.context_factory.reset()

        for date in dates:
            # Get current prices
            try:
                prices_df = self.data_loader.load_prices(symbols, date, date)
                current_prices = prices_df.iloc[0].to_dict()
            except Exception:
                continue

            # Create execution context
            context = self.context_factory.create_context(
                date=date.date(),
                current_prices=current_prices,
                cash=cash
            )

            # Generate signals
            signals = strategy.generate_signals(context, symbols)

            if signals:
                signals_by_date[date.date()] = signals

                # Simulate position updates
                for signal in signals:
                    if signal.action == SignalAction.BUY:

                        cash -= position_size

                    elif signal.action == SignalAction.SELL and signal.symbol in self.context_factory.positions:
                        position_value = (
                            self.context_factory.positions[signal.symbol].quantity *
                            current_prices[signal.symbol]
                        )
                        cash += position_value
                        self.context_factory.update_position(signal, current_prices[signal.symbol])
```

```python
from typing import Dict, Any

        Returns:
            IStrategy instance

        params = params or {}

        # Individual strategies
        if strategy_type == 'rsi':
            return RSIStrategy(

                sell_threshold=params.get('sell_threshold', 70),

                slow=params.get('slow', 26),
                signal=params.get('signal', 9),

            )

        elif strategy_type == 'confluence':
            return ConfluenceStrategy(
                min_agreement=params.get('min_agreement', 2)

        # Composite strategies
```

### 3.2 Signal Generation Detail

```python
# Detailed signal generation loop
for date in dates:
    # Step 1: Get current prices
    try:
        prices_df = self.data_loader.load_prices(symbols, date, date)
        current_prices = prices_df.iloc[0].to_dict()
    except Exception:
        continue

    # Step 2: Calculate portfolio state
    position_value = sum(
        pos.quantity * current_prices.get(pos.symbol, pos.current_price)
        for pos in self.context_factory.positions.values()
    )
    portfolio_value = cash + position_value

    # Step 3: Create execution context
    context = ExecutionContext(
        portfolio_value=portfolio_value,
        available_cash=cash,
        current_positions=list(self.context_factory.positions.values()),
        market_date=date.date(),
        market_status='CLOSED'
    )

    # Step 4: Generate signals (NEW LOGIC - calls strategy)
    signals = strategy.generate_signals(context, symbols)

    # Step 5: Process signals
    if signals:
        signals_by_date[date.date()] = signals

        # Step 6: Simulate position updates
        for signal in signals:
            if signal.action == SignalAction.BUY:
                # Calculate position size (10% of available cash)
                position_size = min(cash * 0.1, cash)
                quantity = position_size / current_prices[signal.symbol]

                # Update position
                self.context_factory.positions[signal.symbol] = PositionState(
                    symbol=signal.symbol,
                    quantity=int(quantity),
                    current_price=current_prices[signal.symbol]
                )
                cash -= position_size

            elif signal.action == SignalAction.SELL:
                if signal.symbol in self.context_factory.positions:
                    pos = self.context_factory.positions[signal.symbol]
                    cash += pos.quantity * current_prices[signal.symbol]
                    del self.context_factory.positions[signal.symbol]
```

```python
import numpy as np
import pandas as pd

class VectorizedRSI:
    """Vectorized RSI calculation for performance optimization"""

    def calculate(self, prices: pd.Series, period: int = 14) -> pd.Series:
        """
        Calculate RSI using vectorized operations.

        Args:
            prices: Price series
            period: RSI period

        Returns:
            RSI values
        """
        # Calculate price changes
        delta = prices.diff()

        # Separate gains and losses
        gains = delta.where(delta > 0, 0)
        losses = -delta.where(delta < 0, 0)

        # Calculate average gains and losses (exponential weighted)
        avg_gains = gains.ewm(span=period, adjust=False).mean()
        avg_losses = losses.ewm(span=period, adjust=False).mean()

        # Calculate RSI
        rs = avg_gains / avg_losses
        rsi = 100 - (100 / (1 + rs))

        return rsi
```

```python
class VectorizedMACD:
    """Vectorized MACD calculation"""

    def calculate(
        self,
        prices: pd.Series,
        fast: int = 12,
        slow: int = 26,
        signal: int = 9
    ) -> tuple[pd.Series, pd.Series]:
        """
        Calculate MACD using vectorized operations.

        Returns:
            (macd_line, signal_line)
        """
        # Calculate EMAs
        ema_fast = prices.ewm(span=fast, adjust=False).mean()
        ema_slow = prices.ewm(span=slow, adjust=False).mean()

        # MACD line
        macd_line = ema_fast - ema_slow

        # Signal line (EMA of MACD)
        signal_line = macd_line.ewm(span=signal, adjust=False).mean()

        return macd_line, signal_line
```

```python
class BatchDataLoader:
    """Load data in batches for performance"""

    def __init__(self, connection, batch_size: int = 50):
        self.conn = connection
        self.batch_size = batch_size

    def load_symbols_batch(
        self,
        symbols: List[str],
        start_date: date,
        end_date: date
    ) -> Dict[str, pd.DataFrame]:
        """
        Load multiple symbols in batch queries.

        Returns:
            Dict mapping symbol to DataFrame
        """
        results = {}

        # Process symbols in batches
        for i in range(0, len(symbols), self.batch_size):
            batch = symbols[i:i + self.batch_size]

            # Single query for entire batch
            query = f"""
            SELECT symbol, date, open, high, low, close, volume
            FROM prices
            WHERE symbol IN ({','.join(['?'] * len(batch))})
              AND date BETWEEN ? AND ?
            ORDER BY symbol, date
            """
            params = batch + [start_date, end_date]

            df = pd.read_sql(query, self.conn, params=params)

            # Split by symbol
            for symbol in batch:
                results[symbol] = df[df['symbol'] == symbol].set_index('date')

        return results
```

```python
from concurrent.futures import ThreadPoolExecutor, as_completed
from typing import Dict, List

class ParallelSignalGenerator:
    """Generate signals in parallel for multiple symbols"""

    def __init__(self, max_workers: int = 4):
        self.max_workers = max_workers

    def generate_signals_parallel(
        self,
        strategy: IStrategy,
        context: ExecutionContext,
        symbols: List[str]
    ) -> List[StrategySignal]:
        """
        Generate signals for multiple symbols in parallel.

        Returns:
            Combined list of signals
        """
        signals = []

        with ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            # Submit tasks
            futures = {
                executor.submit(
                    self._generate_single_symbol_signal,
                    strategy,
                    context,
                    symbol
                ): symbol
                for symbol in symbols
            }

            # Collect results
            for future in as_completed(futures):
                symbol = futures[future]
                try:
                    result = future.result()
                    if result:
                        signals.append(result)
                except Exception as e:
                    print(f"Error generating signal for {symbol}: {e}")

        return signals

        self,
        strategy: IStrategy,
        context: ExecutionContext,
        symbol: str
    ) -> StrategySignal:
        """Generate signal for single symbol"""
        signals = strategy.generate_signals(context, [symbol])
```

```python
class LazyIndicator:
    """Compute indicators only when needed"""

    def __init__(self, compute_fn, prices: pd.Series, **params):
        self.compute_fn = compute_fn
        self.prices = prices
        self.params = params
        self._computed = None

        if self._computed is None:
            self._computed = self.compute_fn(self.prices, **self.params)
        return self._computed

class StrategyWithLazyIndicators(IStrategy):
```

```python
import time
from typing import Dict, List

class PerformanceMonitor:
    """Monitor and report performance metrics"""

    def __init__(self):
        self.metrics: Dict[str, List[float]] = {}

    def measure(self, metric_name: str):
        """Decorator for timing functions"""
        def decorator(func):
            def wrapper(*args, **kwargs):
                start = time.time()
                result = func(*args, **kwargs)
                elapsed = time.time() - start

                if metric_name not in self.metrics:
                    self.metrics[metric_name] = []
                self.metrics[metric_name].append(elapsed)

                return result
            return wrapper
        return decorator

    def report(self) -> Dict[str, Dict[str, float]]:
        """Generate performance report"""
        import numpy as np

        report = {}
        for name, times in self.metrics.items():
            report[name] = {
                'count': len(times),
                'total': sum(times),
                'mean': np.mean(times),
                'median': np.median(times),
                'min': min(times),
                'max': max(times),
                'std': np.std(times)
            }
        return report

    def reset(self):
        """Reset metrics"""
        self.metrics.clear()
```

#### Best Practice 3: Document Confidence Scoring

```python
def _calculate_confidence(self, rsi: float, threshold: float) -> float:
    """
    Calculate confidence based on how far RSI is from threshold.

    Formula: min(1.0, |rsi - threshold| / threshold)

    Args:
        rsi: Current RSI value
        threshold: Buy/sell threshold

    Returns:
        Confidence score 0.0 to 1.0
    """
    if rsi < threshold:  # Buy case
        return min(1.0, (threshold - rsi) / threshold)
    else:  # Sell case
        return min(1.0, (rsi - threshold) / (100 - threshold))
```
