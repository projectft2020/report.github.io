def validate_data_quality(data):
    """
    é©—è­‰æ•¸æ“šå“è³ª
    
    Args:
        data (pd.DataFrame): åƒ¹æ ¼æ•¸æ“š
    
    Returns:
        dict: å“è³ªæª¢æŸ¥çµæœ
    """
    quality_report = {
        'data_shape': data.shape,
        'missing_values': data.isnull().sum().to_dict(),
        'date_range': {
            'start': data.index.min(),
            'end': data.index.max(),
            'total_days': len(data)
        },
        'assets': list(data.columns),
        'data_completeness': {},
        'trading_days_anomaly': {}
    }
    
    # è¨ˆç®—æ¯å€‹è³‡ç”¢çš„æ•¸æ“šå®Œæ•´æ€§
    for asset in data.columns:

        quality_report['data_completeness'][asset] = completeness
        
        # æª¢æŸ¥ç•°å¸¸äº¤æ˜“å¤©æ•¸
        trading_days = len(asset_data.dropna())

            quality_report['trading_days_anomaly'][asset] = {
                'actual': trading_days,
                'expected': expected_days,
                'ratio': trading_days / expected_days
            }
    
    return quality_report
