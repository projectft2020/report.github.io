測量腳本示例:

# scripts/measure_tdd_compliance.py
import subprocess
import re
from collections import defaultdict

def analyze_tdd_compliance(repo_path):
    """分析 TDD 合規性"""
    result = subprocess.run(
        ['git', 'log', '--all', '--oneline', '--reverse'],
        cwd=repo_path,
        capture_output=True,
        text=True
    )

    commits = result.stdout.strip().split('\n')

    test_commits = []
    impl_commits = []

    for commit in commits:
        if 'test' in commit.lower() or 'feat' in commit.lower():
            # 獲取提交的文件
            files = subprocess.run(
                ['git', 'show', '--name-only', '--pretty=format:', commit.split()[0]],
                cwd=repo_path,
                capture_output=True,
                text=True
            ).stdout

            has_test = any('test' in f for f in files.split('\n'))
            has_impl = any('test' not in f and f.endswith('.py') for f in files.split('\n'))

            if has_test:
                test_commits.append(commit)
            if has_impl:
                impl_commits.append(commit)

    compliance = len(test_commits) / max(len(impl_commits), 1)
    return compliance * 100

# scripts/self_assessment.py
import subprocess
import re
from pathlib import Path
from typing import Dict, List
import json

class SelfAssessment:
    """自評工具"""

    def __init__(self, repo_path: str):
        self.repo_path = Path(repo_path)

    def measure_test_coverage(self) -> float:
        """測量測試覆蓋率"""
        result = subprocess.run(
            ['pytest', '--cov=backend', '--cov-report=json'],
            cwd=self.repo_path,
            capture_output=True,
            text=True
        )

        # 解析 coverage.json
        coverage_file = self.repo_path / 'coverage.json'
        if coverage_file.exists():
            with open(coverage_file) as f:
                data = json.load(f)
            return data['totals']['percent_covered']
        return 0.0

    def measure_code_complexity(self) -> Dict[str, float]:
        """測量代碼複雜度"""
        result = subprocess.run(
            ['radon', 'cc', 'backend/', '-a', '-j'],
            cwd=self.repo_path,
            capture_output=True,
            text=True
        )

        if result.returncode == 0:
            data = json.loads(result.stdout)
            # 計算平均複雜度
            total_complexity = 0
            count = 0
            for module in data.values():
                for klass in module.values():
                    for method in klass.values():
                        total_complexity += method['complexity']
                        count += 1

            if count > 0:
                return {'average': total_complexity / count}
        return {'average': 0.0}

    def analyze_tdd_compliance(self) -> Dict[str, float]:
        """分析 TDD 合規性"""
        result = subprocess.run(
            ['git', 'log', '--all', '--oneline', '--since="1 month ago"'],
            cwd=self.repo_path,
            capture_output=True,
            text=True
        )

        commits = result.stdout.strip().split('\n')
        test_commits = 0
        impl_commits = 0

        for commit in commits:
            if not commit:
                continue

            # 獲取提交的文件
            files = subprocess.run(
                ['git', 'show', '--name-only', '--pretty=format:', commit.split()[0]],
                cwd=self.repo_path,
                capture_output=True,
                text=True
            ).stdout

            has_test = any('test' in f for f in files.split('\n'))
            has_impl = any('test' not in f and f.endswith('.py') for f in files.split('\n'))

            if has_test:
                test_commits += 1
            if has_impl:
                impl_commits += 1

        if impl_commits > 0:
            return {
                'test_before_impl': test_commits / impl_commits * 100
            }
        return {'test_before_impl': 0.0}

    def count_pr_reviews(self) -> Dict[str, float]:
        """統計 PR 審查數量"""
        # 這需要根據具體的 Git host (GitHub/GitLab) 實現
        # 這裡是示例
        return {
            'reviews_last_month': 0,
            'avg_response_time_hours': 0
        }

    def generate_report(self) -> Dict:
        """生成自評報告"""
        report = {
            'test_coverage': self.measure_test_coverage(),
            'code_complexity': self.measure_code_complexity(),
            'tdd_compliance': self.analyze_tdd_compliance(),
            'pr_reviews': self.count_pr_reviews()
        }

        return report

    def print_report(self, report: Dict):
        """打印報告"""
        print("=== 自評報告 ===")
        print(f"測試覆蓋率: {report['test_coverage']:.1f}%")
        print(f"平均代碼複雜度: {report['code_complexity']['average']:.1f}")
        print(f"TDD 合規性: {report['tdd_compliance']['test_before_impl']:.1f}%")
        print(f"PR 審查數量: {report['pr_reviews']['reviews_last_month']}")
        print(f"平均回應時間: {report['pr_reviews']['avg_response_time_hours']:.1f} 小時")

# scripts/peer_review.py
import json
from pathlib import Path
from typing import Dict, List
from datetime import datetime

class PeerReviewManager:
    """同行評估管理器"""

    def __init__(self, data_dir: str):
        self.data_dir = Path(data_dir)

    def create_review(self, review: Dict) -> str:
        """創建評估"""
        review_id = f"review-{datetime.now().strftime('%Y%m%d%H%M%S')}.json"
        review_file = self.data_dir / review_id

        with open(review_file, 'w') as f:
            json.dump(review, f, indent=2)

        return review_id

    def load_review(self, review_id: str) -> Dict:
        """加載評估"""
        review_file = self.data_dir / f"{review_id}.json"
        with open(review_file) as f:
            return json.load(f)

    def list_reviews(self, subject: str = None) -> List[Dict]:
        """列出評估"""
        reviews = []
        for review_file in self.data_dir.glob("review-*.json"):
            with open(review_file) as f:
                review = json.load(f)
                if subject is None or review['subject'] == subject:
                    reviews.append(review)
        return reviews

    def calculate_average_score(self, subject: str) -> Dict:
        """計算平均分"""
        reviews = self.list_reviews(subject)

        if not reviews:
            return {}

        # 計算各項平均分
        scores = {}
        for review in reviews:
            for key, value in review.items():
                if isinstance(value, (int, float)) and 0 <= value <= 5:
                    if key not in scores:
                        scores[key] = []
                    scores[key].append(value)

        average = {key: sum(values) / len(values) for key, values in scores.items()}

        return average

    def generate_report(self, subject: str) -> str:
        """生成評估報告"""
        reviews = self.list_reviews(subject)
        average = self.calculate_average_score(subject)

        report = f"""# 同行評估報告

**被評估人:** {subject}
**評估數量:** {len(reviews)}
**評估日期:** {datetime.now().strftime('%Y-%m-%d')}

## 平均得分
"""

        for key, value in average.items():
            report += f"- {key}: {value:.2f} / 5\n"

        report += "\n## 詳細評估\n"
        for review in reviews:
            report += f"\n### 評估人: {review.get('reviewer', '匿名')}\n"
            report += f"- 總評分: {review.get('total_score', 0)} / 5\n"

        return report

if __name__ == '__main__':
    import sys

    manager = PeerReviewManager('./reviews')

    if len(sys.argv) > 1:
        command = sys.argv[1]
        if command == 'report':
            subject = sys.argv[2] if len(sys.argv) > 2 else None
            print(manager.generate_report(subject))
        elif command == 'list':
            subject = sys.argv[2] if len(sys.argv) > 2 else None
            reviews = manager.list_reviews(subject)
            print(f"Found {len(reviews)} reviews")
            for review in reviews:
                print(f"- {review.get('id', 'unknown')}: {review.get('subject', 'unknown')}")

8.3 改進計劃追蹤腳本¶

# scripts/track_improvement.py
import json
from pathlib import Path
from datetime import datetime, timedelta
from typing import Dict, List

class ImprovementTracker:
    """改進計劃追蹤器"""

    def __init__(self, data_dir: str):
        self.data_dir = Path(data_dir)
        self.data_dir.mkdir(exist_ok=True)

    def create_plan(self, plan: Dict) -> str:
        """創建改進計劃"""
        plan_id = f"plan-{datetime.now().strftime('%Y%m%d%H%M%S')}.json"
        plan_file = self.data_dir / plan_id

        with open(plan_file, 'w') as f:
            json.dump(plan, f, indent=2)

        return plan_id

    def load_plan(self, plan_id: str) -> Dict:
        """加載改進計劃"""
        plan_file = self.data_dir / f"{plan_id}.json"
        with open(plan_file) as f:
            return json.load(f)

    def update_progress(self, plan_id: str, task_id: str, status: str, notes: str = ""):
        """更新任務進度"""
        plan = self.load_plan(plan_id)

        for task in plan.get('tasks', []):
            if task['id'] == task_id:
                task['status'] = status
                task['notes'] = notes
                task['updated_at'] = datetime.now().isoformat()
                break

        plan_file = self.data_dir / f"{plan_id}.json"
        with open(plan_file, 'w') as f:
            json.dump(plan, f, indent=2)

    def get_due_tasks(self, plan_id: str) -> List[Dict]:
        """獲取到期任務"""
        plan = self.load_plan(plan_id)
        today = datetime.now()

        due_tasks = []
        for task in plan.get('tasks', []):
            if 'due_date' in task:
                due_date = datetime.fromisoformat(task['due_date'])
                if due_date <= today + timedelta(days=7) and task.get('status') != 'completed':
                    due_tasks.append(task)

        return due_tasks

    def generate_report(self, plan_id: str) -> str:
        """生成進度報告"""
        plan = self.load_plan(plan_id)

        total_tasks = len(plan.get('tasks', []))
        completed_tasks = sum(1 for t in plan.get('tasks', []) if t.get('status') == 'completed')
        progress = completed_tasks / total_tasks * 100 if total_tasks > 0 else 0

        report = f"""# 改進計劃進度報告

**計劃 ID:** {plan_id}
**進度:** {completed_tasks}/{total_tasks} ({progress:.1f}%)
**生成時間:** {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}

## 任務進度
"""

        for task in plan.get('tasks', []):
            status_icon = "✅" if task.get('status') == 'completed' else "⏳"
            report += f"- {status_icon} {task.get('title', '無標題')} - {task.get('status', '無狀態')}\n"
            if 'due_date' in task:
                report += f"  到期日期: {task['due_date']}\n"
            if 'notes' in task:
                report += f"  備註: {task['notes']}\n"

        return report

if __name__ == '__main__':
    import sys

    tracker = ImprovementTracker('./plans')

    if len(sys.argv) > 1:
        command = sys.argv[1]
        if command == 'report':
            plan_id = sys.argv[2] if len(sys.argv) > 2 else None
            print(tracker.generate_report(plan_id))
        elif command == 'due':
            plan_id = sys.argv[2] if len(sys.argv) > 2 else None
            due_tasks = tracker.get_due_tasks(plan_id)
            print(f"Found {len(due_tasks)} due tasks:")
            for task in due_tasks:
                print(f"- {task.get('title', '無標題')} - {task.get('due_date', '無日期')}")

評估體系特點¶

  1. 科學量化: 所有指標都可測量、可驗證
