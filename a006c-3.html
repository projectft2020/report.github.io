from typing import List, Optional

from contextlib import contextmanager

router = APIRouter(prefix="/api/v1/items", tags=["Items"])

# ===== Pydantic Models =====

class ItemBase(BaseModel):
    """項目基礎模型"""
    name: str = Field(..., min_length=1, max_length=100, description="項目名稱")
    description: Optional[str] = Field(None, max_length=1000, description="項目描述")
    price: float = Field(..., gt=0, description="項目價格")

    @validator('name')
    def name_must_not_be_empty(cls, v):
        if not v.strip():
            raise ValueError('name 不能為空')
        return v.strip()

class ItemCreate(ItemBase):
    """創建項目請求模型"""
    pass

class ItemUpdate(ItemBase):
    """更新項目請求模型"""
    name: Optional[str] = Field(None, min_length=1, max_length=100)
    price: Optional[float] = Field(None, gt=0)

class ItemResponse(ItemBase):
    """項目響應模型"""
    id: int
    created_at: datetime
    updated_at: Optional[datetime] = None

    class Config:
        from_attributes = True

class PaginatedResponse(BaseModel):
    """分頁響應模型"""
    items: List[ItemResponse]
    total: int
    page: int
    page_size: int

# ===== Database Connection =====

@contextmanager
def get_db():
    """
    數據庫連接上下文管理器
    自動處理連接的創建和關閉
    """
    conn = duckdb.connect('data.duckdb')

    try:
        # 創建表（如果不存在）
        conn.execute("""
            CREATE TABLE IF NOT EXISTS items (
                id INTEGER PRIMARY KEY,
                name VARCHAR(100) NOT NULL,
                description TEXT,
                price FLOAT NOT NULL CHECK (price > 0),
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                updated_at TIMESTAMP
            )
        """)

        yield conn
    finally:
        conn.close()

    "/{item_id}",
    response_model=ItemResponse,
    summary="更新項目",
    description="根據 ID 更新項目信息"
)

    """更新項目"""
    # 檢查項目是否存在
    existing = db.execute("SELECT id FROM items WHERE id = ?", (item_id,)).fetchone()
    if not existing:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail=f"Item with id {item_id} not found"
        )

    # 構建更新查詢
    update_fields = []
    params = []

    if item.name is not None:
        update_fields.append("name = ?")
        params.append(item.name)

    if item.description is not None:
        update_fields.append("description = ?")
        params.append(item.description)

    if item.price is not None:
        update_fields.append("price = ?")
        params.append(item.price)

    if not update_fields:
        # 沒有字段需要更新
        return await get_item(item_id, db)

    update_fields.append("updated_at = CURRENT_TIMESTAMP")
    params.append(item_id)

    # 執行更新
    result = db.execute(f"""
        UPDATE items
        SET {', '.join(update_fields)}
        WHERE id = ?
        RETURNING id, name, description, price, created_at, updated_at
    """, params).fetchone()

    return ItemResponse(
        id=result[0],
        name=result[1],
        description=result[2],
        price=result[3],
        created_at=result[4],
        updated_at=result[5]
    )

# backend/indicators/simple_rsi.py
"""
RSI (Relative Strength Index) 技術指標實現
使用向量化操作進行高性能計算
"""
import pandas as pd
import numpy as np
from typing import Union

class SimpleRSI:
    """
    相對強弱指標 (RSI) 計算類

    RSI 是一個動量振蕩器，測量價格變動的速度和變化
    範圍：0-100
    通常 >70 為超買，<30 為超賣
    """

    def __init__(self, period: int = 14):
        """
        初始化 RSI 指標

        Args:
            period: RSI 週期，默認 14
        """
        if period <= 0:
            raise ValueError("Period must be positive")

        self.period = period
        self._cache = {}

    def calculate(self, prices: Union[pd.Series, np.ndarray]) -> pd.Series:
        """
        計算 RSI 指標

        Args:
            prices: 價格序列（可以是 pandas Series 或 numpy array）

        Returns:
            RSI 值序列（pandas Series）

        Raises:
            ValueError: 如果輸入數據不足
        """
        # 轉換為 pandas Series
        if isinstance(prices, np.ndarray):
            prices = pd.Series(prices)

        # 檢查數據長度
        if len(prices) < self.period:
            raise ValueError(
                f"Insufficient data: need at least {self.period} data points, "
                f"got {len(prices)}"
            )

        # 計算價格變化
        delta = prices.diff()

        # 分離漲跌幅
        gains = delta.where(delta > 0, 0.0)
        losses = -delta.where(delta < 0, 0.0)

        # 計算平均漲跌幅（使用指數加權移動平均）
        avg_gains = gains.ewm(span=self.period, adjust=False).mean()
        avg_losses = losses.ewm(span=self.period, adjust=False).mean()

        # 計算相對強度 (RS)
        # 避免除以零
        avg_losses = avg_losses.replace(0, np.nan)
        rs = avg_gains / avg_losses

        # 計算 RSI
        rsi = 100 - (100 / (1 + rs))

        return rsi

    def get_signals(
        self,
        prices: Union[pd.Series, np.ndarray],
        buy_threshold: float = 30.0,
        sell_threshold: float = 70.0
    ) -> pd.DataFrame:
        """
        根據 RSI 生成交易信號

        Args:
            prices: 價格序列
            buy_threshold: 買入閾值（超賣），默認 30
            sell_threshold: 賣出閾值（超買），默認 70

        Returns:
            包含 RSI 值和信號的 DataFrame
        """
        rsi = self.calculate(prices)

        # 生成信號
        signals = pd.DataFrame({
            'price': prices,
            'rsi': rsi
        })

        # 買入信號：RSI 從下方突破買入閾值
        signals['buy'] = (rsi < buy_threshold) & (rsi.shift(1) >= buy_threshold)

        # 賣出信號：RSI 從上方突破賣出閾值
        signals['sell'] = (rsi > sell_threshold) & (rsi.shift(1) <= sell_threshold)

        return signals

    def get_current_rsi(self, prices: Union[pd.Series, np.ndarray]) -> float:
        """
        獲取最新的 RSI 值

        Args:
            prices: 價格序列

        Returns:
            最新的 RSI 值
        """
        rsi = self.calculate(prices)
        return rsi.iloc[-1]

    def is_overbought(self, prices: Union[pd.Series, np.ndarray], threshold: float = 70.0) -> bool:
        """
        檢查是否超買

        Args:
            prices: 價格序列
            threshold: 超買閾值

        Returns:
            如果 RSI > 閾值則返回 True
        """
        rsi = self.calculate(prices)
        return rsi.iloc[-1] > threshold

    def is_oversold(self, prices: Union[pd.Series, np.ndarray], threshold: float = 30.0) -> bool:
        """
        檢查是否超賣

        Args:
            prices: 價格序列
            threshold: 超賣閾值

        Returns:
            如果 RSI < 閾值則返回 True
        """
        rsi = self.calculate(prices)
        return rsi.iloc[-1] < threshold

# backend/indicators/simple_macd.py
"""
MACD (Moving Average Convergence Divergence) 技術指標實現
使用向量化操作進行高性能計算
"""
import pandas as pd
import numpy as np
from typing import Union, Tuple

class SimpleMACD:
    """
    移動平均收斂發散指標 (MACD) 計算類

    MACD 是一個趨勢跟隨動量指標
    顯示兩條移動平均線之間的關係

    組成部分：
    - MACD 線: 快速 EMA - 慢速 EMA
    - 信號線: MACD 的 EMA
    - 柱狀圖: MACD - 信號線
    """

    def __init__(
        self,
        fast_period: int = 12,
        slow_period: int = 26,
        signal_period: int = 9
    ):
        """
        初始化 MACD 指標

        Args:
            fast_period: 快速 EMA 週期，默認 12
            slow_period: 慢速 EMA 週期，默認 26
            signal_period: 信號線 EMA 週期，默認 9
        """
        if fast_period >= slow_period:
            raise ValueError("fast_period must be less than slow_period")

        if signal_period <= 0:
            raise ValueError("signal_period must be positive")

        self.fast_period = fast_period
        self.slow_period = slow_period
        self.signal_period = signal_period
        self._cache = {}

    def calculate(
        self,
        prices: Union[pd.Series, np.ndarray]
    ) -> Tuple[pd.Series, pd.Series, pd.Series]:
        """
        計算 MACD 指標

        Args:
            prices: 價格序列

        Returns:
            (macd_line, signal_line, histogram) 元組
            - macd_line: MACD 線
            - signal_line: 信號線
            - histogram: 柱狀圖 (MACD - 信號線)
        """
        # 轉換為 pandas Series
        if isinstance(prices, np.ndarray):
            prices = pd.Series(prices)

        # 檢查數據長度
        min_period = max(self.slow_period, self.signal_period)
        if len(prices) < min_period:
            raise ValueError(
                f"Insufficient data: need at least {min_period} data points, "
                f"got {len(prices)}"
            )

        # 計算 EMA
        ema_fast = prices.ewm(span=self.fast_period, adjust=False).mean()
        ema_slow = prices.ewm(span=self.slow_period, adjust=False).mean()

        # 計算 MACD 線
        macd_line = ema_fast - ema_slow

        # 計算信號線（MACD 的 EMA）
        signal_line = macd_line.ewm(span=self.signal_period, adjust=False).mean()

        # 計算柱狀圖
        histogram = macd_line - signal_line

        return macd_line, signal_line, histogram

    def get_signals(
        self,
        prices: Union[pd.Series, np.ndarray]
    ) -> pd.DataFrame:
        """
        根據 MACD 生成交易信號

        Args:
            prices: 價格序列

        Returns:
            包含 MACD 值和信號的 DataFrame

        信號規則：
        - 買入信號：MACD 線從下方突破信號線（金叉）
        - 賣出信號：MACD 線從上方突破信號線（死叉）
        """
        macd_line, signal_line, histogram = self.calculate(prices)

        # 生成信號
        signals = pd.DataFrame({
            'price': prices,
            'macd': macd_line,
            'signal': signal_line,
            'histogram': histogram
        })

        # 金叉：MACD 向上突破信號線
        signals['buy'] = (
            (macd_line > signal_line) &
            (macd_line.shift(1) <= signal_line.shift(1))
        )

        # 死叉：MACD 向下突破信號線
        signals['sell'] = (
            (macd_line < signal_line) &
            (macd_line.shift(1) >= signal_line.shift(1))
        )

        return signals

    def get_current_macd(
        self,
        prices: Union[pd.Series, np.ndarray]
    ) -> Tuple[float, float, float]:
        """
        獲取最新的 MACD 值

        Args:
            prices: 價格序列

        Returns:
            (macd, signal, histogram) 元組
        """
        macd_line, signal_line, histogram = self.calculate(prices)

        return (
            macd_line.iloc[-1],
            signal_line.iloc[-1],
            histogram.iloc[-1]
        )

    def is_bullish_crossover(
        self,
        prices: Union[pd.Series, np.ndarray]
    ) -> bool:
        """
        檢查是否有看漲交叉（金叉）

        Args:
            prices: 價格序列

        Returns:
            如果有金叉則返回 True
        """
        signals = self.get_signals(prices)
        return signals['buy'].iloc[-1]

    def is_bearish_crossover(
        self,
        prices: Union[pd.Series, np.ndarray]
    ) -> bool:
        """
        檢查是否有看跌交叉（死叉）

        Args:
            prices: 價格序列

        Returns:
            如果有死叉則返回 True
        """
        signals = self.get_signals(prices)
        return signals['sell'].iloc[-1]

    def is_bullish_trend(
        self,
        prices: Union[pd.Series, np.ndarray]
    ) -> bool:
        """
        檢查是否處於看漲趨勢

        Args:
            prices: 價格序列

        Returns:
            如果 MACD > 信號線則返回 True
        """
        macd, signal, _ = self.get_current_macd(prices)
        return macd > signal

    def is_bearish_trend(
        self,
        prices: Union[pd.Series, np.ndarray]
    ) -> bool:
        """
        檢查是否處於看跌趨勢

        Args:
            prices: 價格序列

        Returns:
            如果 MACD < 信號線則返回 True
        """
        macd, signal, _ = self.get_current_macd(prices)
        return macd < signal

/**

 * @param {Function} fetcher - 數據獲取函數
 * @param {Object} options - 選項配置

export const useApiCache = (queryKey, fetcher, options = {}) => {
  const {
    staleTime = cachePresets.user.staleTime,
    gcTime = cachePresets.user.gcTime,
    enabled = true,
    retry = 1,
    retryDelay = attemptIndex => Math.min(1000 * 2 ** attemptIndex, 30000),
    refetchOnWindowFocus = false,
    refetchOnMount = true,
    ...restOptions
  } = options;

  return useQuery({
    queryKey,
    queryFn: fetcher,
    staleTime,
    gcTime,
    enabled,
    retry,
    retryDelay,
    refetchOnWindowFocus,
    refetchOnMount,
    ...restOptions,
  });
};

/**
 * 用於執行變更操作的 Hook
 *
 * @param {Function} mutationFn - 變更函數
 * @param {Object} options - 選項配置
 * @returns {Object} 變更結果對象
 *
 * @example
 * const { mutate, isLoading, error } = useApiMutation(
 *   (data) => fetch('/api/v1/items', {
 *     method: 'POST',
 *     body: JSON.stringify(data)
 *   }).then(r => r.json()),
 *   {
 *     invalidateKeys: [['items']],
 *     onSuccess: () => toast.success('Created successfully!')
 *   }
 * );
 */
export const useApiMutation = (mutationFn, options = {}) => {
  const queryClient = useQueryClient();

  const {
    invalidateKeys = [],
    onSuccess,
    onError,
    showToast = true,
    successMessage = 'Operation successful',
    errorMessage = 'Operation failed',
    ...restOptions
  } = options;

  return useMutation({
    mutationFn,
    onSuccess: (data, variables, context) => {
      // 使相關查詢失效
      invalidateKeys.forEach(key => {
        queryClient.invalidateQueries(key);
      });

      // 顯示成功通知
      if (showToast) {
        toast.success(successMessage);
      }

      // 調用成功回調
      if (onSuccess) {
        onSuccess(data, variables, context);
      }
    },
    onError: (error, variables, context) => {
      console.error('Mutation error:', error);

      // 顯示錯誤通知
      const message = error.response?.data?.detail ||
                      error.response?.data?.message ||
                      error.message ||
                      errorMessage;

      if (showToast) {
        toast.error(message);
      }

      // 調用錯誤回調
      if (onError) {
        onError(error, variables, context);
      }
    },
    ...restOptions,
  });
};

/**
 * 批量變更操作 Hook
 *
 * @param {Function} mutationFn - 變更函數
 * @param {Object} options - 選項配置
 * @returns {Object} 變更結果對象
 */
export const useBatchMutation = (mutationFn, options = {}) => {
  const { mutateAsync, ...mutationResult } = useApiMutation(mutationFn, options);

  const batchMutate = async (items, concurrency = 5) => {
    const results = [];
    const errors = [];

    for (let i = 0; i < items.length; i += concurrency) {
      const batch = items.slice(i, i + concurrency);
      const promises = batch.map(async (item) => {
        try {
          const result = await mutateAsync(item);
          return { success: true, data: result, item };
        } catch (error) {
          return { success: false, error, item };
        }
      });

      const batchResults = await Promise.all(promises);
      batchResults.forEach(result => {
        if (result.success) {
          results.push(result.data);
        } else {
          errors.push(result);
        }
      });
    }

    return { results, errors };
  };

  return {

"""
策略引擎基礎接口和類型定義
"""
from dataclasses import dataclass
from enum import Enum
from typing import List
from datetime import date
from abc import ABC, abstractmethod

class SignalAction(Enum):
    """交易信號動作"""
    BUY = "BUY"
    SELL = "SELL"
    HOLD = "HOLD"

@dataclass
class Signal:
    """
    交易信號

    Attributes:
        symbol: 標的代碼
        action: 交易動作
        confidence: 信心分數 (0.0 to 1.0)
        reason: 信號原因說明
    """
    symbol: str
    action: SignalAction
    confidence: float
    reason: str = ""

    def __post_init__(self):
        """驗證信心分數"""
        if not (0.0 <= self.confidence <= 1.0):
            raise ValueError(f"Confidence must be 0.0-1.0, got {self.confidence}")

@dataclass
class ExecutionContext:
    """
    策略執行上下文

    Attributes:
        portfolio_value: 投資組合總值
        available_cash: 可用現金
        current_positions: 當前持倉列表
        market_date: 市場日期
        market_status: 市場狀態 ('OPEN', 'CLOSED', etc.)
    """
    portfolio_value: float
    available_cash: float
    current_positions: List['PositionState']
    market_date: date
    market_status: str

@dataclass
class PositionState:
    """
    持倉狀態

    Attributes:
        symbol: 標的代碼
        quantity: 持倉數量
        current_price: 當前價格
        entry_price: 入場價格（可選）
        unrealized_pnl: 未實現盈虧（可選）
    """
    symbol: str
    quantity: int
    current_price: float
    entry_price: float = None
    unrealized_pnl: float = None

class IStrategy(ABC):
    """
    策略接口

    所有策略必須實現此接口
    """

    @property
    @abstractmethod
    def name(self) -> str:
        """
        策略標識符

        Returns:
            策略名稱
        """
        raise NotImplementedError

    @abstractmethod
    def generate_signals(
        self,
        context: ExecutionContext,
        symbols: List[str]
    ) -> List[Signal]:
        """
        生成交易信號（核心方法）

        約束：
        - 返回信號列表（每個標的最多一個）
        - 只返回 BUY/SELL（HOLD 通過不返回表示）
        - 信心分數 0.0-1.0
        - 對相同輸入必須確定性
        - 交易前檢查 market_status
        - 尊重持倉狀態和現金約束

        Args:
            context: 執行上下文
            symbols: 要分析的標的列表

        Returns:
            交易信號列表
        """
        raise NotImplementedError

# backend/strategies/rsi_strategy.py
"""
RSI 均值回歸策略
"""
from .base import IStrategy, Signal, SignalAction, ExecutionContext
from typing import List
from datetime import date, timedelta
import pandas as pd
import pandas_ta as ta

class RSIStrategy(IStrategy):
    """
    RSI 均值回歸策略

    交易邏輯：
    - 當 RSI < 買入閾值時，生成買入信號
    - 當 RSI > 賣出閾值時，生成賣出信號

    注意：
    - 只在 market_status == 'CLOSED' 時生成信號
    - 買入信號僅在未持倉時生成
    - 賣出信號僅在持倉時生成
    """

    name = "rsi_mean_reversion"

    def __init__(
        self,
        period: int = 14,
        buy_threshold: float = 30.0,
        sell_threshold: float = 70.0,
        min_trade_size: float = 1000.0
    ):
        """
        初始化 RSI 策略

        Args:
            period: RSI 週期
            buy_threshold: 買入閾值（超賣）
            sell_threshold: 賣出閾值（超買）
            min_trade_size: 最小交易金額
        """
        if not (0 < buy_threshold < sell_threshold < 100):
            raise ValueError("Invalid threshold values")

        if period <= 0:
            raise ValueError("Period must be positive")

        if min_trade_size <= 0:
            raise ValueError("Min trade size must be positive")

        self.period = period
        self.buy_threshold = buy_threshold
        self.sell_threshold = sell_threshold
        self.min_trade_size = min_trade_size
        self.rsi_cache = {}

    def generate_signals(
        self,
        context: ExecutionContext,
        symbols: List[str]
    ) -> List[Signal]:
        """生成交易信號"""
        signals = []

        for symbol in symbols:
            # 檢查市場狀態
            if context.market_status != 'CLOSED':
                continue

            # 檢查是否已持有
            existing_position = next(
                (p for p in context.current_positions if p.symbol == symbol),
                None
            )

            # 獲取 RSI 值
            try:
                rsi = self._get_rsi(symbol, context.market_date)

                if pd.isna(rsi):
                    continue

                # 生成買入信號（僅當未持有時）
                if rsi < self.buy_threshold and not existing_position:
                    confidence = self._calculate_confidence(rsi, self.buy_threshold)

                    if confidence >= 0.5:  # 只在信心高於 50% 時交易
                        signals.append(Signal(
                            symbol=symbol,
                            action=SignalAction.BUY,
                            confidence=confidence,
                            reason=f"RSI ({rsi:.2f}) below buy threshold ({self.buy_threshold})"
                        ))

                # 生成賣出信號（僅當持有时）
                elif rsi > self.sell_threshold and existing_position:
                    confidence = self._calculate_confidence(rsi, self.sell_threshold)

                    if confidence >= 0.5:
                        signals.append(Signal(
                            symbol=symbol,
                            action=SignalAction.SELL,
                            confidence=confidence,
                            reason=f"RSI ({rsi:.2f}) above sell threshold ({self.sell_threshold})"
                        ))

            except Exception as e:
                # 記錄錯誤但繼續處理其他標的
                print(f"Error processing {symbol}: {e}")
                continue

        return signals

    def _get_rsi(self, symbol: str, date: date) -> float:
        """
        獲取 RSI 值並緩存

        Args:
            symbol: 標的代碼
            date: 日期

        Returns:
            RSI 值
        """
        cache_key = f"{symbol}_{date}"

        if cache_key not in self.rsi_cache:
            # 使用 pandas-ta 計算 RSI
            # 這裡需要從數據庫獲取價格數據
            prices = self._load_prices(symbol, date - timedelta(days=self.period * 3), date)

            if len(prices) < self.period:
                return float('nan')

            rsi_values = ta.momentum.RSIIndicator(
                prices['close'],
                window=self.period
            ).rsi()

            self.rsi_cache[cache_key] = rsi_values.iloc[-1] if len(rsi_values) > 0 else float('nan')

        return self.rsi_cache[cache_key]

    def _load_prices(self, symbol: str, start_date: date, end_date: date) -> pd.DataFrame:
        """
        從數據庫加載價格數據

        Args:
            symbol: 標的代碼
            start_date: 開始日期
            end_date: 結束日期

        Returns:
            價格數據 DataFrame
        """
        # 實際實現會從數據庫查詢
        # 這裡是簡化版本，實際使用時需要實現
        import duckdb

        conn = duckdb.connect('data.duckdb')

        query = """
            SELECT date, close
            FROM prices
            WHERE symbol = ?
              AND date BETWEEN ? AND ?
            ORDER BY date
        """

        try:
            df = pd.read_sql(query, conn, params=[symbol, start_date, end_date])
            return df
        finally:
            conn.close()

    def _calculate_confidence(self, rsi: float, threshold: float) -> float:
        """
        根據距離閾值的距離計算信心分數

        Args:
            rsi: RSI 值
            threshold: 閾值

        Returns:
            信心分數 (0.0 to 1.0)
        """
        if rsi < threshold:  # 買入情況
            return min(1.0, (threshold - rsi) / threshold)
        else:  # 賣出情況
            return min(1.0, (rsi - threshold) / (100 - threshold))

# backend/strategies/composite_strategy.py
"""
複合策略 - 組合多個策略
"""
from .base import IStrategy, Signal, SignalAction, ExecutionContext
from typing import List, Dict, Callable

class CompositeStrategy(IStrategy):
    """
    複合策略

    組合多個子策略，使用指定的合併算法生成信號
    """

    name = "composite_strategy"

    # 支持的合併算法
    MERGE_ALGORITHMS = {
        'majority_vote': self._merge_majority_vote,
        'unanimous': self._merge_unanimous,
        'weighted': self._merge_weighted,
        'any': self._merge_any,
    }

    def __init__(
        self,
        strategies: List[IStrategy],
        merge_algorithm: str = 'majority_vote',
        weights: List[float] = None
    ):
        """
        初始化複合策略

        Args:
            strategies: 子策略列表
            merge_algorithm: 合併算法 ('majority_vote', 'unanimous', 'weighted', 'any')
            weights: 加權投票的權重（僅用於 'weighted' 算法）
        """
        if not strategies:
            raise ValueError("At least one strategy is required")

        if merge_algorithm not in self.MERGE_ALGORITHMS:
            raise ValueError(f"Invalid merge algorithm: {merge_algorithm}")

        if merge_algorithm == 'weighted' and (not weights or len(weights) != len(strategies)):
            raise ValueError("Weights must be provided and match number of strategies")

        self.strategies = strategies
        self.merge_algorithm = merge_algorithm
        self.weights = weights or [1.0] * len(strategies)

    def generate_signals(
        self,
        context: ExecutionContext,
        symbols: List[str]
    ) -> List[Signal]:
        """生成交易信號"""
        # 收集所有策略的信號
        all_signals_by_symbol = {symbol: [] for symbol in symbols}

        for strategy in self.strategies:
            try:
                strategy_signals = strategy.generate_signals(context, symbols)

                for signal in strategy_signals:
                    all_signals_by_symbol[signal.symbol].append(signal)
            except Exception as e:
                print(f"Error in strategy {strategy.name}: {e}")
                continue

        # 合併每個標的的信號
        merged_signals = []
        for symbol, signals in all_signals_by_symbol.items():
            if not signals:
                continue

            merged_signal = self._merge_signals(symbol, signals)
            if merged_signal:
                merged_signals.append(merged_signal)

        return merged_signals

    def _merge_signals(self, symbol: str, signals: List[Signal]) -> Signal:
        """使用配置的算法合併多個信號"""
        merge_fn = self.MERGE_ALGORITHMS[self.merge_algorithm]
        return merge_fn(self, symbol, signals)

    def _merge_majority_vote(self, symbol: str, signals: List[Signal]) -> Signal:
        """
        多數決合併

        信號數量最多的動作獲勝
        """
        buy_count = sum(1 for s in signals if s.action == SignalAction.BUY)
        sell_count = sum(1 for s in signals if s.action == SignalAction.SELL)

        if buy_count > sell_count:
            avg_confidence = (
                sum(s.confidence for s in signals if s.action == SignalAction.BUY)
                / buy_count
            )
            reasons = [s.reason for s in signals if s.action == SignalAction.BUY]
            return Signal(
                symbol=symbol,
                action=SignalAction.BUY,
                confidence=avg_confidence,
                reason=f"Majority buy ({buy_count}/{len(signals)}): {', '.join(reasons)}"
            )
        elif sell_count > buy_count:
            avg_confidence = (
                sum(s.confidence for s in signals if s.action == SignalAction.SELL)
                / sell_count
            )
            reasons = [s.reason for s in signals if s.action == SignalAction.SELL]
            return Signal(
                symbol=symbol,
                action=SignalAction.SELL,
                confidence=avg_confidence,
                reason=f"Majority sell ({sell_count}/{len(signals)}): {', '.join(reasons)}"
            )

        return None  # 平局

    def _merge_unanimous(self, symbol: str, signals: List[Signal]) -> Signal:
        """
        一致同意合併

        所有信號必須是相同的動作
        """
        if not signals:
            return None

        actions = [s.action for s in signals]

        if all(a == SignalAction.BUY for a in actions):
            reasons = [s.reason for s in signals]
            avg_confidence = sum(s.confidence for s in signals) / len(signals)
            return Signal(
                symbol=symbol,
                action=SignalAction.BUY,
                confidence=avg_confidence,
                reason=f"Unanimous buy: {', '.join(reasons)}"
            )
        elif all(a == SignalAction.SELL for a in actions):
            reasons = [s.reason for s in signals]
            avg_confidence = sum(s.confidence for s in signals) / len(signals)
            return Signal(
                symbol=symbol,
                action=SignalAction.SELL,
                confidence=avg_confidence,
                reason=f"Unanimous sell: {', '.join(reasons)}"
            )

        return None

    def _merge_weighted(self, symbol: str, signals: List[Signal]) -> Signal:
        """
        加權投票合併

        使用權重計算每個動作的加權分數
        """
        if not signals:
            return None

        # 創建信號到權重的映射
        signal_weights = {id(s): w for s, w in zip(signals, self.weights)}

        buy_score = 0.0
        sell_score = 0.0
        buy_weight_sum = 0.0
        sell_weight_sum = 0.0

        for signal in signals:
            weight = signal_weights[id(signal)]

            if signal.action == SignalAction.BUY:
                buy_score += signal.confidence * weight
                buy_weight_sum += weight
            elif signal.action == SignalAction.SELL:
                sell_score += signal.confidence * weight
                sell_weight_sum += weight

        # 比較加權分數
        if buy_weight_sum > 0 and buy_score > sell_score:
            avg_confidence = buy_score / buy_weight_sum
            return Signal(
                symbol=symbol,
                action=SignalAction.BUY,
                confidence=avg_confidence,
                reason=f"Weighted buy (score: {buy_score:.2f})"
            )
        elif sell_weight_sum > 0 and sell_score > buy_score:
            avg_confidence = sell_score / sell_weight_sum
            return Signal(
                symbol=symbol,
                action=SignalAction.SELL,
                confidence=avg_confidence,
                reason=f"Weighted sell (score: {sell_score:.2f})"
            )

        return None

    def _merge_any(self, symbol: str, signals: List[Signal]) -> Signal:
        """
        任一信號

        任何一個信號即觸發
        """
        if not signals:
            return None

        # 返回第一個信號
        return signals[0]

# backend/strategies/factory.py
"""
策略工廠
集中管理策略的註冊和創建
"""
from typing import Dict, Type, Any
from .base import IStrategy
from .rsi_strategy import RSIStrategy
from .macd_strategy import MACDStrategy
from .composite_strategy import CompositeStrategy

class StrategyFactory:
    """
    策略工廠

    提供策略的註冊、創建和列表功能
    """

    _strategies: Dict[str, Type[IStrategy]] = {}

    @classmethod
    def register(cls, strategy_type: str, strategy_class: Type[IStrategy]):
        """
        註冊策略類

        Args:
            strategy_type: 策略類型標識符
            strategy_class: 策略類

        Raises:
            ValueError: 如果策略類型已存在
        """
        if strategy_type in cls._strategies:
            raise ValueError(f"Strategy type {strategy_type} already registered")

        cls._strategies[strategy_type] = strategy_class

    @classmethod
    def unregister(cls, strategy_type: str):
        """
        註銷策略類

        Args:
            strategy_type: 策略類型標識符
        """
        if strategy_type in cls._strategies:
            del cls._strategies[strategy_type]

    @classmethod
    def create(cls, strategy_type: str, params: Dict[str, Any] = None) -> IStrategy:
        """
        創建策略實例

        Args:
            strategy_type: 策略類型標識符
            params: 策略參數

        Returns:
            策略實例

        Raises:
            ValueError: 如果策略類型不存在
        """
        params = params or {}

        if strategy_type not in cls._strategies:
            raise ValueError(f"Unknown strategy type: {strategy_type}")

        strategy_class = cls._strategies[strategy_type]

        try:
            return strategy_class(**params)
        except TypeError as e:
            raise ValueError(f"Invalid parameters for {strategy_type}: {e}")

    @classmethod
    def list_strategies(cls) -> list:
        """
        列出所有已註冊的策略類型

        Returns:
            策略類型列表
        """
        return list(cls._strategies.keys())

    @classmethod
    def is_registered(cls, strategy_type: str) -> bool:
        """
        檢查策略類型是否已註冊

        Args:
            strategy_type: 策略類型標識符

        Returns:
            如果已註冊返回 True
        """
        return strategy_type in cls._strategies

    def _check_test_before_impl(self) -> Dict:
        """檢查測試先於實現"""
        # 分析 Git 提交歷史
        result = subprocess.run(
            ['git', 'log', '--all', '--oneline', '--reverse'],
            cwd=self.repo_path,
            capture_output=True,
            text=True
        )

        commits = result.stdout.strip().split('\n')
        test_commits = []
        impl_commits = []

        for commit in commits:
            if not commit:
                continue

            # 獲取提交的文件
            files = subprocess.run(
                ['git', 'show', '--name-only', '--pretty=format:', commit.split()[0]],
                cwd=self.repo_path,
                capture_output=True,
                text=True
            ).stdout

            has_test = any('test' in f for f in files.split('\n'))
            has_impl = any('test' not in f and f.endswith('.py') for f in files.split('\n'))

            if has_test:
                test_commits.append(commit)
            if has_impl:
                impl_commits.append(commit)

        if impl_commits > 0:
            compliance = len(test_commits) / len(impl_commits)
        else:
            compliance = 0

        # 階段 1 標準: ≥ 80%
        passed = compliance >= 0.8

        return {
            'status': 'PASS' if passed else 'FAIL',
            'test_commits': len(test_commits),
            'impl_commits': len(impl_commits),
            'compliance': compliance,
            'score': min(100, (compliance / 0.8) * 100)
        }

Q1: 如何開始第一個項目？¶

A: 使用 Conventional Commits 格式：

feat(items): add pagination support

- Add skip and limit parameters to list endpoint
- Update frontend to display pagination controls

Closes #123

Q4: 如何調試 API 錯誤？¶

6.4 架構師 FAQ¶

Q1: 如何設計微服務架構？¶

A: 1. 領域驅動設計: 根據業務邊界拆分服務 2. 單一職責: 每個服務只做一件事 3. 通信方式: 選擇合適的通信方式（REST, gRPC, 消息隊列） 4. 數據獨立: 每個服務有自己的數據庫 5. 服務發現: 使用 Consul 或 Kubernetes Service 6. API 網關: 統一入口，處理認證、限流、路由 7. 監控和日誌: 集中化的監控和日誌系統

Q2: 如何部署到 Kubernetes？¶

A: 1. 容器化: 每個服務打包成 Docker 鏡像 2. 編寫 Manifests: Deployment, Service, ConfigMap, Secret 3. 使用 Helm: 打包和模板化 K8s 資源 4. 配置 Ingress: 暴露服務到外部 5. 設置 HPA: 自動擴縮容 6. 配置監控: Prometheus + Grafana 7. 配置日誌: Loki + Promtail 8. CI/CD: 自動構建和部署流水線

Q3: 如何集成機器學習模型？¶

A: 1. 模型選擇: 根據問題類型選擇合適的模型 2. 特征工程: 數據預處理和特征提取 3. 模型訓練: 使用交叉驗證和超參優化 4. 模型評估: 使用適當的評估指標 5. 模型序列化: 使用 joblib 或 MLflow 6. API 封裝: 使用 FastAPI 提供 REST 接口 7. 模型監控: 監控性能、數據漂移、模型衰減 8. A/B 測試: 在生產環境中驗證模型

Q4: 如何制定技術路線圖？¶

A: 1. 現狀分析: 評估當前技術狀況 2. 識別技術債: 列出需要償還的技術債 3. 設定目標: 短期（3 個月）、中期（6 個月）、長期（1 年） 4. 優先級排序: 根據業務價值和風險排序 5. 資源規劃: 預估時間和人力 6. 里程碑: 設置明確的里程碑 7. 溝通和審查: 定期與團隊和利益相關者溝通 8. 迭代調整: 根據實際情況調整路線圖

Q5: 如何培養團隊？¶

A: 1. 導師計劃: 為新成員分配導師 2. 代碼審查: 通過代碼審查傳授最佳實踐 3. 知識分享: 定期的技術分享會 4. 培訓: 提供內部培訓和外部培訓機會 5. 挑戰性項目: 分配有挑戰性的項目促進成長 6. 反饋: 定期提供建設性的反饋 7. 職業發展: 討論職業目標和發展路徑 8. 技術規範: 制定和執行編碼規範和架構規範
