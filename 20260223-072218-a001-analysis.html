Sub-Agent Output File Loss Mechanism Analysis¶

Task ID: 20260223-072218-a001 Agent: Charlie Analyst Status: completed Timestamp: 2026-02-23T07:22:18Z

Executive Summary¶

The sub-agent output file loss mechanism is a critical reliability gap where task status (reported as "completed") and actual output file existence become desynchronized. The root cause lies in a fragmented output pipeline: sub-agents must write both an output file AND a status file, but there is no atomic guarantee that both operations succeed together. This creates a false completion signal that wastes investigation time and erodes trust in the system. The failure manifests when a sub-agent reports completion via session announce or writes a status file, but the actual output file is never created, never written, or fails to be copied to the main workspace.

Analysis¶

System Architecture Overview¶

The sub-agent output pipeline consists of three critical components:

1. Agent Output Skill Protocol
   ├─ Write output file: /workspace-[agent]/outputs/[task_id].md
   └─ Write status file: /workspace-[agent]/outputs/.status/[task_id].status

2. Session Announce Mechanism
   └─ Sub-agent reports completion to main session

3. Task Sync Process (task_sync.py)
   ├─ Scan for .status files
   ├─ Update tasks.json status based on .status
   └─ Copy output files to main workspace

Point 1: Status File Written, Output File Missing (CRITICAL)¶

Failure Scenario:

1. Implement Atomic Write Operations (Priority: HIGH)¶

Solution:

def write_output_atomically(task_id: str, content: str, output_path: str, agent_type: str):
    """
    Atomic write: ensures both output file and status file are written
    or neither is written.

    Returns: (success: bool, error: str | None)
    """
    workspace = Path.home() / f'.openclaw/workspace-{agent_type}'
    output_file = workspace / 'outputs' / f"{task_id}.md"
    status_dir = workspace / 'outputs/.status'
    status_file = status_dir / f"{task_id}.status"

    try:
        # 1. Write output file first
        output_file.write_text(content, encoding='utf-8')

        # 2. Validate output file exists and is not empty
        if not output_file.exists() or output_file.stat().st_size == 0:
            return False, "Output file is empty or not created"

        # 3. Calculate checksum
        import hashlib
        checksum = hashlib.md5(content.encode('utf-8')).hexdigest()

        # 4. Create status directory
        status_dir.mkdir(parents=True, exist_ok=True)

        # 5. Write status file
        status_data = {
            "task_id": task_id,
            "status": "completed",
            "output_path": str(output_file),
            "completed_at": datetime.now(timezone.utc).isoformat(),
            "exit_code": 0,
            "agent_type": agent_type,
            "checksum": checksum,  # NEW: Validate integrity
            "file_size": len(content)  # NEW: Validate size
        }
        status_file.write_text(json.dumps(status_data, indent=2))

        return True, None

    except Exception as e:
        # Rollback: delete any partially created files
        if output_file.exists():
            output_file.unlink()
        if status_file.exists():
            status_file.unlink()

        return False, str(e)

Benefits: - Guarantees both files or neither - Validates output file before creating status - Rollback on any failure - Adds integrity validation (checksum, size)

def announce_completion(task_id: str, output_path: str):
    """
    Announce completion only after verifying file exists.
    """
    output_file = Path(output_path).expanduser()

    # CRITICAL: Verify file exists before announcing
    if not output_file.exists():
        return f"[ERROR] Task {task_id} failed: Output file not created at {output_path}"

    # CRITICAL: Verify file is not empty
    if output_file.stat().st_size == 0:
        return f"[ERROR] Task {task_id} failed: Output file is empty"

    # CRITICAL: Verify file is readable
    try:
        content = output_file.read_text(encoding='utf-8')
        if len(content) < 100:  # Sanity check
            return f"[ERROR] Task {task_id} failed: Output file too small ({len(content)} bytes)"
    except Exception as e:
        return f"[ERROR] Task {task_id} failed: Cannot read output file: {e}"

    # Only announce if all checks pass
    return f"Analysis complete. Output: {output_path}"

Benefits: - Prevents false announcements - Validates file before reporting - Provides clear error messages - Catches write failures early

3. Implement Retry and Rollback in Task Sync (Priority: HIGH)¶

def copy_output_file_with_retry(output_path: str, task_id: str, max_retries: int = 3):
    """
    Copy output file with retry mechanism.
    If copy fails, ROLLBACK task status to pending.

    Returns: (success: bool, error: str | None)
    """
    from time import sleep

    output_file = Path(output_path).expanduser()

    # CRITICAL: Validate source file exists
    if not output_file.exists():
        return False, f"Source file not found: {output_path}"

    # Get destination
    tasks = load_tasks()
    task = next((t for t in tasks if t.get('id') == task_id), None)
    if not task:
        return False, f"Task {task_id} not found"

    project_id = task.get('project_id', 'unknown')
    main_workspace = Path.home() / '.openclaw/workspace/kanban/projects'
    target_dir = main_workspace / project_id
    target_file = target_dir / f"{task_id}.md"

    target_dir.mkdir(parents=True, exist_ok=True)

    # Retry loop
    for attempt in range(1, max_retries + 1):
        try:
            shutil.copy2(output_file, target_file)

            # Verify copy succeeded
            if not target_file.exists():
                raise Exception("Target file not created after copy")

            # Verify file sizes match
            if target_file.stat().st_size != output_file.stat().st_size:
                raise Exception(f"File size mismatch: source={output_file.stat().st_size}, target={target_file.stat().st_size}")

            logger.info(f"[Sync] ✓ Copied output to: {target_file} (attempt {attempt})")
            return True, None

        except Exception as e:
            logger.warning(f"[Sync] Copy attempt {attempt} failed: {e}")
            if attempt < max_retries:
                sleep(2 ** attempt)  # Exponential backoff: 2s, 4s, 8s
            else:
                # ALL RETRIES FAILED - ROLLBACK
                logger.error(f"[Sync] All {max_retries} copy attempts failed for task {task_id}")

                # CRITICAL: Rollback task status to pending
                update_task_status(task_id, 'pending', {
                    'copy_error': str(e),
                    'rolled_back_at': datetime.now(timezone.utc).isoformat(),
                    'retry_count': task.get('retry_count', 0) + 1
                })

                return False, f"Copy failed after {max_retries} attempts: {e}"

    return False, "Unknown error"

def pre_flight_write_check(output_path: str, content_size: int) -> (bool, str | None):
    """
    Check if we can successfully write to target path.

    Returns: (can_write: bool, error: str | None)
    """
    output_file = Path(output_path).expanduser()

    # Check parent directory exists
    parent_dir = output_file.parent
    if not parent_dir.exists():
        try:
            parent_dir.mkdir(parents=True, exist_ok=True)
        except Exception as e:
            return False, f"Cannot create parent directory: {e}"

    # Check write permissions
    test_file = parent_dir / f".write_test_{os.getpid()}.tmp"
    try:
        test_file.touch()
        test_file.unlink()
    except Exception as e:
        return False, f"Write permission denied: {e}"

    # Check disk space (minimum 10x content size)
    stat = os.statvfs(str(parent_dir))
    free_space = stat.f_bavail * stat.f_frsize
    if free_space < content_size * 10:
        return False, f"Insufficient disk space: need {content_size * 10}, have {free_space}"

    return True, None

import fcntl

def write_with_lock(file_path: Path, content: str):
    """
    Write file with exclusive lock to prevent race conditions.
    """
    with open(file_path, 'w') as f:
        # Acquire exclusive lock (non-blocking)
        try:
            fcntl.flock(f.fileno(), fcntl.LOCK_EX | fcntl.LOCK_NB)
        except IOError:
            raise Exception("File is locked by another process")

        try:
            f.write(content)
        finally:
            fcntl.flock(f.fileno(), fcntl.LOCK_UN)

Benefits: - Prevents concurrent write issues - Avoids corruption from race conditions - Safe for multi-process environments

def validate_file_integrity(file_path: Path, expected_checksum: str = None, expected_size: int = None):
    """
    Validate file integrity after copy.

    Returns: (valid: bool, error: str | None)
    """
    if not file_path.exists():
        return False, "File does not exist"

    # Check size
    actual_size = file_path.stat().st_size
    if expected_size and actual_size != expected_size:
        return False, f"Size mismatch: expected {expected_size}, got {actual_size}"

    if actual_size == 0:
        return False, "File is empty"

    # Check checksum
    if expected_checksum:
        import hashlib
        with open(file_path, 'rb') as f:
            actual_checksum = hashlib.md5(f.read()).hexdigest()

        if actual_checksum != expected_checksum:
            return False, f"Checksum mismatch: expected {expected_checksum}, got {actual_checksum}"

    # Try to read as text
    try:
        content = file_path.read_text(encoding='utf-8')
        if len(content) < 50:  # Sanity check
            return False, f"File too small: {len(content)} bytes"
    except Exception as e:
        return False, f"Cannot read file: {e}"

    return True, None
