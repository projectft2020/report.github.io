        bh.end_date,

def get_backtest_by_id(history_id: int, user_id: int) -> Optional[Dict]:
    query = """

        bh.*,

        br.total_trades
    FROM backtest_history bh
    LEFT JOIN backtest_runs br ON bh.run_id = br.id

def compare_backtests(history_ids: List[int], user_id: int) -> List[Dict]:
    placeholders = ','.join([f':id_{i}' for i in range(len(history_ids))])
    params = {"user_id": user_id}
    params.update({f"id_{i}": hist_id for i, hist_id in enumerate(history_ids)})

    query = f"""
    SELECT
        bh.*,
        br.win_rate,
        br.profit_factor,
        br.total_trades
    FROM backtest_history bh
    LEFT JOIN backtest_runs br ON bh.run_id = br.id
    WHERE bh.id IN ({placeholders}) AND bh.user_id = :user_id
    ORDER BY bh.created_at DESC
    """

    results = db.execute(query, params).fetchall()

    comparisons = []
    for row in results:
        comparison = {
            "id": row.id,
            "user_id": row.user_id,
            "run_id": row.run_id,
            "strategy_id": row.strategy_id,
            "params": row.params,
            "start_date": row.start_date.isoformat() if row.start_date else None,
            "end_date": row.end_date.isoformat() if row.end_date else None,
            "created_at": row.created_at.isoformat(),
            "status": row.status,
            "win_rate": float(row.win_rate) if row.win_rate else None,
            "profit_factor": float(row.profit_factor) if row.profit_factor else None,
            "total_trades": row.total_trades
        }
        comparisons.append(comparison)

    return comparisons

import React, { useState, useEffect } from 'react';

function BacktestHistoryPage() {
  const [histories, setHistories] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    async function fetchHistories() {
      try {
        setLoading(true);
        const response = await fetch('/api/backtest/history');
        if (!response.ok) {
          throw new Error('Failed to fetch backtest history');
        }
        const data = await response.json();
        setHistories(data.histories);
      } catch (err) {
        setError(err.message);
      } finally {
        setLoading(false);
      }
    }

    fetchHistories();
  }, []);

  const formatWinRate = (winRate) => {
    if (winRate === null || winRate === undefined) {
      return 'N/A';
    }
    return `${winRate.toFixed(2)}%`;
  };

  const formatProfitFactor = (profitFactor) => {
    if (profitFactor === null || profitFactor === undefined) {
      return 'N/A';
    }
    if (profitFactor === Infinity) {
      return '∞';
    }
    return profitFactor.toFixed(2);
  };

  if (loading) {
    return <div className="loading">Loading backtest history...</div>;
  }

  if (error) {
    return <div className="error">Error: {error}</div>;
  }

  return (
    <div className="backtest-history-page">
      <h1>Backtest History</h1>

      <div className="history-stats">
        <div className="stat-card">
          <h3>Total Records</h3>
          <p>{histories.length}</p>
        </div>
      </div>

      <div className="history-table-container">
        <table className="history-table">
          <thead>
            <tr>
              <th>Date</th>
              <th>Strategy ID</th>
              <th>Status</th>
              <th>Win Rate</th>
              <th>Profit Factor</th>
              <th>Total Trades</th>
              <th>Actions</th>
            </tr>
          </thead>
          <tbody>
            {histories.map(history => (
              <tr key={history.id}>
                <td>{new Date(history.created_at).toLocaleDateString()}</td>
                <td>{history.strategy_id}</td>
                <td>
                  <span className={`status status-${history.status.toLowerCase()}`}>
                    {history.status}
                  </span>
                </td>
                <td className="metric">
                  {formatWinRate(history.win_rate)}
                </td>
                <td className="metric">
                  {formatProfitFactor(history.profit_factor)}
                </td>
                <td>{history.total_trades || 'N/A'}</td>
                <td>
                  <div className="action-buttons">
                    <button 
                      onClick={() => viewDetails(history.id)}
                      className="btn-view"
                    >
                      View
                    </button>
                    <button 
                      onClick={() => compareHistory(history.id)}
                      className="btn-compare"
                    >
                      Compare
                    </button>
                  </div>
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>

      {histories.length === 0 && (
        <div className="no-data">
          <p>No backtest history found. Run a backtest to see results here.</p>
        </div>
      )}
    </div>
  );
}

import React, { useState, useEffect } from 'react';

function BacktestComparison({ historyIds }) {
  const [comparisons, setComparisons] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function fetchComparisons() {
      try {
        const response = await fetch('/api/backtest/compare', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
          },
          body: JSON.stringify({ history_ids: historyIds }),
        });

        if (!response.ok) throw new Error('Failed to fetch comparisons');
        const data = await response.json();
        setComparisons(data.comparisons);
      } catch (error) {
        console.error('Error fetching comparisons:', error);
      } finally {
        setLoading(false);
      }
    }

    if (historyIds.length > 0) {
      fetchComparisons();
    }
  }, [historyIds]);

  const formatWinRate = (winRate) => {
    if (winRate === null || winRate === undefined) return 'N/A';
    return `${winRate.toFixed(2)}%`;
  };

  const formatProfitFactor = (profitFactor) => {
    if (profitFactor === null || profitFactor === undefined) return 'N/A';
    if (profitFactor === Infinity) return '∞';
    return profitFactor.toFixed(2);
  };

  if (loading) return <div>Loading comparisons...</div>;

  return (
    <div className="backtest-comparison">
      <h2>Backtest Comparison</h2>

      <table className="comparison-table">
        <thead>
          <tr>
            <th>Strategy</th>
            <th>Date</th>
            <th>Win Rate</th>
            <th>Profit Factor</th>
            <th>Total Trades</th>
          </tr>
        </thead>
        <tbody>
          {comparisons.map(comp => (
            <tr key={comp.id}>
              <td>{comp.strategy_id}</td>
              <td>{new Date(comp.created_at).toLocaleDateString()}</td>
              <td className="metric">
                {formatWinRate(comp.win_rate)}
              </td>
              <td className="metric">
                {formatProfitFactor(comp.profit_factor)}
              </td>
              <td>{comp.total_trades || 'N/A'}</td>
            </tr>
          ))}
        </tbody>
      </table>

      <div className="comparison-summary">
        <h3>Performance Summary</h3>
        <div className="summary-metrics">
          {comparisons.length > 0 && (
            <>
              <div className="summary-item">
                <label>Avg Win Rate:</label>
                <span>
                  {formatWinRate(
                    comparisons.reduce((sum, c) => sum + (c.win_rate || 0), 0) / comparisons.length
                  )}
                </span>
              </div>
              <div className="summary-item">
                <label>Avg Profit Factor:</label>
                <span>
                  {formatProfitFactor(
                    comparisons.reduce((sum, c) => sum + (c.profit_factor || 0), 0) / comparisons.length
                  )}
                </span>
              </div>
            </>
          )}
        </div>
      </div>
    </div>
  );
}

export default BacktestComparison;

from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.orm import Session
from typing import List, Optional
from services.history_service import (
    list_backtest_history,
    get_backtest_by_id,
    compare_backtests
)

router = APIRouter()

@router.get("/api/backtest/history")
async def get_backtest_history(
    user_id: int,
    limit: int = 100,
    offset: int = 0,
    db: Session = Depends(get_db)
):
    """獲取回測歷史記錄列表"""
    try:
        histories = list_backtest_history(user_id)
        return {"histories": histories[offset:offset + limit]}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@router.get("/api/backtest/history/{history_id}")
async def get_backtest_history_detail(
    history_id: int,
    user_id: int,
    db: Session = Depends(get_db)
):
    """獲取單個回測歷史記錄詳情"""
    try:
        history = get_backtest_by_id(history_id, user_id)
        if not history:
            raise HTTPException(status_code=404, detail="History not found")
        return {"history": history}
    except HTTPException:
        raise
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@router.post("/api/backtest/compare")
async def compare_backtest_history(
    request: dict,
    user_id: int,
    db: Session = Depends(get_db)
):
    """比較多個回測歷史記錄"""
    try:
        history_ids = request.get("history_ids", [])
        if not history_ids:
            raise HTTPException(status_code=400, detail="No history IDs provided")

        comparisons = compare_backtests(history_ids, user_id)
        return {"comparisons": comparisons}
    except HTTPException:
        raise
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
