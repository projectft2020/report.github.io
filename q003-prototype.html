        # å‰µå»ºæ–°åƒæ•¸

                if np.random.random() < mutation_rate:

                    variation = value * 0.1 * (np.random.random() * 2 - 1)
                    new_params[key] = value + variation

        new_hypothesis = Hypothesis(
            core_statement=hypothesis.core_statement,
            strategy_category=hypothesis.strategy_category,
            parameters=new_params,
            confidence=max(0.5, min(0.9, hypothesis.confidence * (1 + np.random.normal(0, 0.1)))
        )

        return new_hypothesis

class MockBacktester:
    """æ¨¡æ“¬å›æ¸¬å™¨ï¼ˆç°¡åŒ–ç‰ˆï¼Œä¸ä½¿ç”¨çœŸå¯¦æ•¸æ“šï¼‰"""
    def __init__(self, base_sharpe=1.0, base_mdd=0.15):
        self.base_sharpe = base_sharpe
        self.base_mdd = base_mdd

    def backtest(
        self,
        hypothesis: Hypothesis
    ) -> BacktestResult:
        """
        åŸ·è¡Œæ¨¡æ“¬å›æ¸¬

        æ³¨æ„ï¼šé€™æ˜¯æ¨¡æ“¬å›æ¸¬ï¼Œçµæœä¸åæ˜ çœŸå¯¦å¸‚å ´è¡¨ç¾
        """
        # æ ¹æ“šåƒæ•¸ç”Ÿæˆæ¨¡æ“¬çµæœ
        params = hypothesis.parameters

        # åŸºæ–¼åƒæ•¸å’Œç­–ç•¥é¡å‹ç”Ÿæˆåˆç†çš„å›å ±
        if hypothesis.strategy_category == "momentum":
            # å‹•é‡ç­–ç•¥ï¼šè¼ƒé«˜çš„å¤æ™®æ¯”ç‡ï¼Œè¼ƒé«˜çš„å›æ’¤
            sharpe = self.base_sharpe + np.random.normal(0, 0.3)
            mdd = self.base_mdd + np.random.normal(0, 0.05)
        elif hypothesis.strategy_category == "mean_reversion":
            # å‡å€¼å›æ­¸ï¼šè¼ƒç©©å®šçš„å¤æ™®æ¯”ç‡ï¼Œè¼ƒä½çš„å›æ’¤
            sharpe = self.base_sharpe - 0.2 + np.random.normal(0, 0.2)
            mdd = self.base_mdd * 0.8 + np.random.normal(0, 0.03)
        elif hypothesis.strategy_category == "rsi":
            # RSIï¼šä¸­ç­‰çš„å¤æ™®æ¯”ç‡ï¼Œä¸­ç­‰çš„å›æ’¤
            sharpe = self.base_sharpe - 0.1 + np.random.normal(0, 0.25)
            mdd = self.base_mdd * 0.9 + np.random.normal(0, 0.04)
        else:
            sharpe = self.base_sharpe + np.random.normal(0, 0.4)
            mdd = self.base_mdd + np.random.normal(0, 0.08)

        # åŸºæ–¼ä¿¡å¿ƒåº¦èª¿æ•´
        sharpe *= hypothesis.confidence
        mdd /= hypothesis.confidence

        # è¨ˆç®—å…¶ä»–æŒ‡æ¨™
        annual_return = sharpe * 0.15  # å‡è¨­å¹´åŒ–æ³¢å‹•ç‡ 15%
        total_return = (1 + annual_return) ** 3 - 1  # 3 å¹´
        win_rate = 0.5 + (sharpe - 1.0) * 0.05  # åŸºæ–¼å¤æ™®æ¯”ç‡çš„å‹ç‡
        win_rate = max(0.4, min(0.7, win_rate))  # é™åˆ¶åœ¨åˆç†ç¯„åœ
        n_trades = np.random.randint(20, 100)  # äº¤æ˜“æ¬¡æ•¸

        return BacktestResult(
            total_return=total_return,
            annual_return=annual_return,
            sharpe_ratio=max(0, sharpe),
            max_drawdown=min(0.5, max(0, mdd)),
            win_rate=win_rate,
            n_trades=n_trades
        )

class SimpleFeatureMap:
    """ç°¡åŒ–çš„ç‰¹å¾µåœ–ç³»çµ±"""
    def __init__(self, n_bins: int = 5):
        """
        åˆå§‹åŒ–ç‰¹å¾µåœ–

        Args:
            n_bins: æ¯å€‹ç¶­åº¦çš„åˆ†ç®±æ•¸é‡
        """
        self.n_bins = n_bins
        self.grid = {}  # æ ¼å¼: {(bin_x, bin_y): [strategies]}
        self.strategies_per_cell = 3  # æ¯å€‹å–®å…ƒæ ¼æœ€å¤§ç­–ç•¥æ•¸

    def extract_features(
        self,
        strategy: Strategy
    ) -> np.ndarray:
        """
        æå–ç‰¹å¾µå‘é‡

        Args:
            strategy: ç­–ç•¥å°è±¡

        Returns:
            ç‰¹å¾µå‘é‡
        """
        # ç‰¹å¾µ 1: ç­–ç•¥é¡åˆ¥ï¼ˆ0: momentum, 1: mean_reversion, 2: rsiï¼‰
        category_map = {"momentum": 0, "mean_reversion": 1, "rsi": 2}
        feature1 = category_map.get(strategy.hypothesis.strategy_category, 0)

        # ç‰¹å¾µ 2: å¤æ™®æ¯”ç‡ï¼ˆæ­¸ä¸€åŒ–åˆ° [0, 1]ï¼‰
        sharpe = strategy.backtest_result.sharpe_ratio
        feature2 = min(1.0, max(0.0, sharpe / 3.0))

        # ç‰¹å¾µ 3: æœ€å¤§å›æ’¤ï¼ˆæ­¸ä¸€åŒ–åˆ° [0, 1]ï¼Œå€¼è¶Šå°è¶Šå¥½ï¼‰
        mdd = strategy.backtest_result.max_drawdown
        feature3 = min(1.0, max(0.0, 1.0 - mdd / 0.5))

        # ç‰¹å¾µ 4: å‹ç‡
        win_rate = strategy.backtest_result.win_rate
        feature4 = win_rate

        # ç‰¹å¾µ 5: ä¿¡å¿ƒåº¦
        confidence = strategy.hypothesis.confidence
        feature5 = confidence

        return np.array([feature1, feature2, feature3, feature4, feature5])

    def discretize_features(
        self,
        features: np.ndarray
    ) -> Tuple[int, int]:
        """
        å°‡ç‰¹å¾µé›¢æ•£åŒ–åˆ°ç¶²æ ¼åæ¨™

        Args:
            features: ç‰¹å¾µå‘é‡

        Returns:
            (x, y) ç¶²æ ¼åæ¨™
        """
        # ä½¿ç”¨å‰å…©å€‹ç‰¹å¾µä½œç‚ºç¶²æ ¼åæ¨™
        # ç‰¹å¾µ 1: ç­–ç•¥é¡åˆ¥
        x = int(features[0] * (self.n_bins - 1) / 2)

        # ç‰¹å¾µ 2: å¤æ™®æ¯”ç‡
        y = int(features[1] * (self.n_bins - 1))

        return (x, y)

    def add_strategy(self, strategy: Strategy):
        """
        æ·»åŠ ç­–ç•¥åˆ°ç‰¹å¾µåœ–

        Args:
            strategy: ç­–ç•¥å°è±¡
        """
        # æå–ç‰¹å¾µ
        features = self.extract_features(strategy)

        # é›¢æ•£åŒ–
        cell_coords = self.discretize_features(features)

        # æ·»åŠ åˆ°ç¶²æ ¼
        if cell_coords not in self.grid:
            self.grid[cell_coords] = []

        # æª¢æŸ¥å®¹é‡
        if len(self.grid[cell_coords]) >= self.strategies_per_cell:
            # å®¹é‡å·²æ»¿ï¼Œç§»é™¤æœ€å·®çš„ç­–ç•¥
            self.grid[cell_coords].sort(
                key=lambda s: s.fitness,
                reverse=True
            )
            if strategy.fitness > self.grid[cell_coords][-1].fitness:
                self.grid[cell_coords][-1] = strategy
        else:
            self.grid[cell_coords].append(strategy)

    def get_diverse_strategies(
        self,
        n: int,
        exclude: List[str] = None
    ) -> List[Strategy]:
        """
        ç²å–å¤šæ¨£åŒ–çš„ç­–ç•¥

        Args:
            n: éœ€è¦çš„ç­–ç•¥æ•¸é‡
            exclude: è¦æ’é™¤çš„ç­–ç•¥ ID åˆ—è¡¨

        Returns:
            å¤šæ¨£åŒ–ç­–ç•¥åˆ—è¡¨
        """
        if exclude is None:
            exclude = []

        diverse_strategies = []
        visited_cells = list(self.grid.keys())
        np.random.shuffle(visited_cells)

        for cell_coords in visited_cells:
            strategies_in_cell = [
                s for s in self.grid.get(cell_coords, [])
                if s.id not in exclude
            ]

            if strategies_in_cell:
                # é¸æ“‡æœ€ä½³ç­–ç•¥
                best_strategy = max(
                    strategies_in_cell,
                    key=lambda s: s.fitness
                )
                diverse_strategies.append(best_strategy)
                visited_cells.remove(cell_coords)

            if len(diverse_strategies) >= n:
                break

        return diverse_strategies

    def get_summary(self) -> str:
        """
        ç²å–ç‰¹å¾µåœ–æ‘˜è¦

        Returns:
            æ‘˜è¦å­—ç¬¦ä¸²
        """
        total_strategies = sum(len(strategies) for strategies in self.grid.values())
        total_cells = len(self.grid)
        filled_cells = sum(
            1 for strategies in self.grid.values()
            if strategies
        )

        return (
            f"FeatureMap: {total_strategies} strategies in "
            f"{filled_cells}/{total_cells} cells "
            f"({filled_cells/total_cells*100:.1f}% filled)"
        )

def calculate_fitness(
    result: BacktestResult,
    weights: dict = None
) -> float:
    """
    è¨ˆç®—é©æ‡‰åº¦

    Args:
        result: å›æ¸¬çµæœ
        weights: å„æŒ‡æ¨™æ¬Šé‡

    Returns:
        é©æ‡‰åº¦åˆ†æ•¸
    """
    if weights is None:
        # é»˜èªæ¬Šé‡
        weights = {
            'sharpe_ratio': 0.4,
            'max_drawdown': -0.3,  # è² æ¬Šé‡ï¼Œè¶Šå°è¶Šå¥½
            'win_rate': 0.2,
            'total_return': 0.1
        }

    # è¨ˆç®—åŠ æ¬Šåˆ†æ•¸
    fitness = (
        weights['sharpe_ratio'] * result.sharpe_ratio +
        weights['max_drawdown'] * result.max_drawdown +
        weights['win_rate'] * result.win_rate +
        weights['total_return'] * min(1.0, result.total_return / 2.0)
    )

    return fitness
